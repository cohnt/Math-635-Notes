\thm{
	Let $q=\exp_{p}(t,v)$, $\norm{v}=1$ be a conjugate point of $p$. Then $\forall{}t_{2}>t_{1}$, $t\mapsto\exp_{p}(tv)$ is not minimizing on $[0,t_{2}]$.\nn
	Proof: By the hypothesis, there's a Jacobi field $J$ f $\gamma$ such that $J\ne{}0$, $J(0)=0$, and $J(t_{1})=0$. We will construct a variation of $\gamma$ on $[0,t_{2}]$ with $E''<0$. Define
	\[
		\tilde{J}(t)=\left\{\begin{array}{ll}
			J(t) & 0\le{}t\le{}t_{1}\\
			0 & t_{1}\le{}t\le{}t_{2}
		\end{array}\right.
	\]
	Because $J(t_{1})=0$, this variation is continuous at $t_{1}$, so it's clearly continuous on $[0,t_{2}]$. Let $W\in\Gamma_{\gamma}(TM)$ be smooth, supported near $t_{1}$, and defined such that $W(t_{1})=\Delta\tilde{J}'(t_{1})\ne{}0$. It's nonzero because $\Delta\tilde{J}'(t_{1})=0$ would imply that $J'(t_{1})=0$, which would mean $J=0$, a contradiction with our original assumption. Now, we define the actual variation we're going to use. Let
	\[
		V_{\varepsilon}=\tilde{J}+\varepsilon{}W
	\]
	for some small $0<\varepsilon\ll{}1$. This is a proper variation of $\gamma$ on $[0,t_{2}]$. Now compute $E''(0)$ (associated with $V_{\varepsilon}$).
	\[
		E''(0)=I(V_{\varepsilon},V_{\varepsilon})=I(\tilde{J}+\varepsilon{}W,\tilde{J}+\varepsilon{}W)=I(\tilde{J},\tilde{J})+2\varepsilon{}I(\tilde{J},W)+\varepsilon^{2}I(W,W)
	\]
	where $I$ is the bilinear form defined previously. Well,
	\[
		I(\tilde{J},\tilde{J})=-\int_{0}^{t_{2}}\iprod{\tilde{J},\cancelto{0}{\ptxt{Jacobi operator on $\tilde{J}$}}}-\left<\vphantom{\tilde{J}}\right.\underbrace{\tilde{J}(t_{1})}_{=0},\Delta\tilde{J}'(t_{1})\left.\vphantom{\tilde{J}}\right>\,dt=0
	\]
	\[
		I(\tilde{J},W)=\int_{0}^{t_{2}}\iprod{\tilde{J}',W}-\iprod{\mc{R}(\tilde{J},\dot\gamma)\dot\gamma,W}\,dt
	\]
	We use integration by parts, with $\frac{d}{dt}\iprod{W,\tilde{J}'}=\iprod{W',\tilde{J}'}+\iprod{W,\tilde{J}''}$, to compute
	\[
		\int_{0}^{t_{2}}\iprod{\tilde{J}',W}\,dt=-\int_{0}^{t_{2}}\iprod{W,\tilde{J}''}\,dt-\iprod{W(t_{1}),\Delta\tilde{J}'(t_{1})}
	\]
	Now, we combine with the $\iprod{\mc{R}(\tilde{J},\dot\gamma)\dot\gamma,W}$ term. Using the fact that $\tilde{J}$ satisfies the Jacobi equation, they cancel, and we're left with
	\[
		I(\tilde{J},W)=-\iprod{W(t_{1}),\Delta\tilde{J}'(t_{1})}=-\norm{\Delta\tilde{J}'(t_{1})}^{2}<0
	\]
	Thus, $E''(0)=\varepsilon^{2}I(W,W)-2\varepsilon\norm{\Delta\tilde{J}'(t_{1})}^{2}$. So for $\varepsilon\ll{}1$, $E''(0)<$, so for $s$ small enough, $t$-curves in a variation of $\gamma$ with $\tilde{V}_{\varepsilon}$ are shorter than $\gamma$.\proven
}

\section*{Completeness}

\par\noindent
(Chpater 7 in Do Carmo)\n

\defn{
	$M$ is \u{geodesically complete} iff $\forall{}p\in{}M$, $\exp_{p}$ is defined on all of $T_{p}M$.\n
}

\ex{
	If $M$ is compact, $M$ is geodesically complete.\nn
	Why? Well, if $M$ is compact, then the unit tangent bundle $TM_{\ones}=\set{(p,v)\in{}TM:\norm{v}=1}$ is compact. So geodesic flow is given by the flow of a certain field on $TM_{\ones}$ (up to scaling by time), and smooth vector fields on compact manifolds are complete.\proven
}

\defn{
	$M$ is complete iff $(M,d)$ is a complete metric space.\n
}

\par\noindent
The \b{big idea} we're working towards is
\thm{
	(Hopf-Rinow) $M$ is geodesically complete iff $M$ is a complete metric space.\n
}

\thm{
	Let $M$ be connected. Let $p\in{}M$ such that $\exp_{p}$ is defined on all of $T_{p}M$. Fix $q\in{}M$. Then there's a geodesic $\gamma$ from $p$ to $q$, and $d(p,q)=l(\gamma)$.\nn
	Proof: Let $\varepsilon>0$ be such that there's a geodesic sphere $S_{\varepsilon}$ of radius $\varepsilon$ centered at $p$. Let $p'\in{}S_{\varepsilon}$ be a point minimizing the map
	\[
		\map{S_{\varepsilon}}{\R}{x}{d(x,q)}
	\]
	That is, $p'$ is the point on $S_{\varepsilon}$ which is closest to $q$. By compactness, $p'$ exists, and $p'=\exp_{p}(\varepsilon{}v)$ for some $v\in{}T_{p}M$ with $\norm{v}=1$. Now, we want to show $\exp_{p}(d(p,q)v)=q$...\n
}

\lemma{
	$d(p,q)=\underbrace{d(p,p')}_{=\varepsilon}+d(p',q)$.\n
	Proof: $\le$ is just a direct application of the triangle inequality. For $\ge$, let $c$ be any path from $p$ to $q$, and let $w$ be the point where $c$ intersects $S_{\varepsilon}$. Then $l(c)=l(\overarc{pw})+l(\overarc{wq})\ge\varepsilon+d(p',q)$. Now, take the infimum over all such paths $c$. We get
	\[
		d(p,q)=\underset{c}{\inf{}}l(c)\ge\varepsilon+d(p',q)=d(p,p')+d(p',q)
	\]
	\proven
}

\par\noindent
Returning to the proof of the theorem, introduce $\mathscr{T}\eqdef\set{t\in[0,d(p,q)]\mid{}d(p,q)=t+d(\gamma(t),q)}$. We observe the following facts about $\mathscr{T}$:
\begin{itemize}
	\item $\mathscr{T}\ne\emptyset$ because $\varepsilon\in\mathscr{T}$ by the lemma.
	\item $\mathscr{T}$ is closed, because it's the preimage of a closed set under a continuous function.
	\item $\forall{}t\in\mathscr{T}$, $d(\gamma(t),p)=t$.
\end{itemize}
We want to show $d(p,q)=\sup\mathscr{T}$. We will argue this by contradiction: assume $t_{1}\eqdef\sup\mathscr{T}<d(p,q)$. Then $t_{1}+\delta<d(p,q)$. $S_{\delta}$ exists centered at $\gamma(t)$, so then we'll show $t+\delta\in\mathscr{T}$, thus contradicting the definition of $t_{1}$ as the supremum of $\mathscr{T}$. We will do this next time.
