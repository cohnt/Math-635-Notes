\section*{Sectional Curvature}

\defn{
	$\forall{}p\in{}M$, $\forall{}u,v\in{}T_{p}M$ linearly independent, the \u{sectional curvature} is defined to be
	\[
		K_{p}(u,v)\eqdef\frac{\mc{R}(u,v,v,u)}{\abs{u\wedge{}v}^{2}}=\frac{\iprod{\mc{R}(u,v)v,u}}{\norm{u}^{2}\norm{v}^{2}-\iprod{u,v}^{2}}
	\]
	\n
}

\par\noindent
Last time, we claimed that $K_{p}(u,v)$ only depends on the plane $\pi=\vspan(u,v)\in\Gr(2,T_{p}M)$. So really, $K_{p}$ is a function $K_{p}:\Gr(2,T_{p}M)\to\R$.\n

\par\noindent
Proof of this claim: We want to show that the right-hand side is invariant under the following ``moves'':
\begin{itemize}[itemsep=0pt, topsep=0pt]
	\item $(u,v)\rightsquigarrow(\lambda{}u,v)$ (for $\lambda\ne{}0$)
	\item $(u,v)\rightsquigarrow(v,u)$
	\item $(u,v)\rightsquigarrow(u+\lambda{}v,v)$
\end{itemize}
This is easy to do using known properties of $\mc{R}$.\n

\par\noindent
Observe: If $\dim{}M=2$, then think about $\mc{R}$ and its symmetries. How many independent components does it have? Well, for $\mc{R}(u,v,u',v')$ (with $u$ and $v$ linearly independent), we can write $(u',v')$ in terms of $(u,v)$ via a change of basis:
\[
	\begin{bmatrix}
		u'\\
		v'
	\end{bmatrix}=\underbrace{\begin{bmatrix}
		a_{11} & a_{12}\\
		a_{21} & a_{22}
	\end{bmatrix}}_{A}\begin{bmatrix}
		u\\
		v
	\end{bmatrix}
\]
So $\mc{R}(u,v,u',v')=\det(A)\mc{R}(u,v,u,v)=-\det(A)\mc{R}(u,v,v,u)$. So we conclude that in dimension $2$, there's only one degree of freedom! All the information that's contained in $\mc{R}$ reduces to knowing the function $K(p)=\mc{R}_{p}(e,f,f,e)$, where $(e,f)$ is an orthonormal basis of $T_{p}M$ ($K:M\to\R$).\n

\prop{
	$K$ from above is the Gaussian curvature.\n
}

\par\noindent
In dimension $n$, there are $\frac{n^{2}(n^{2}-1)}{12}$ degrees of freedom.\n

\prop{
	The function $\displaystyle{}K:\bigcup_{p\in{}M}\Gr(2,T_{p}M)\to\R$ completely determines $\mc{R}$.\nn
	Proof: This is just an algebraic exercise. As a preliminary, start with a bilinear map $b:V\times{}V\to\R$. Note that $b$ is the sum of a symmetric and antisymmetric bilinear form; this can be seen by writing $b(x,y)=x\tpose{}My$ for a unique matrix $M$, and then noting that we can write $M=\frac{1}{2}(M+M\tpose)+\frac{1}{2}(M-M\tpose)$. $M+M\tpose$ is symmetric, and $M-M\tpose$ is antisymmetric.\nn
	Now, if $b$ is symmetric, then $b$ is uniquely determined by the mapping $x\mapsto{}q(x)=b(x,x)$, because $q(x+y)=q(x)+q(y)+2b(x,y)$. We can then solve for $b(x,y)$. (This is known as the ``polarization identity''.)\nn
	\b{Cor:} If $b(x,x)=0$, $\forall{}x\in{}V$, then $b$ must be skew-symmetric.\nn
	We now continue the proof of the proposition. Let $V$ be a vector space (e.g. $T_{p}M$). Let $\mc{R},\mc{R}':V\times{}V\times{}V\times{}V\to\R$ be multilinear maps, with the symmetries of the Riemannian curvature. Then define $D=\mc{R}-\mc{R}'$, and it has the same symmetries. Assume $\forall{}v,w,x$, $D(v,w,w,v)=0$ ($x$ will be used later). We want to show $D\equiv{}0$.\nn
	Why is this true? Well,
	\begin{align*}
		? & =D(v+w,x,x,v+w)\\
		& =\underbrace{D(v,x,x,v)}_{=0}+D(v,x,x,w)+D(w,x,x,v)+\underbrace{D(w,x,x,w)}_{=0}\\
		& =D(v,x,x,w)-D(v,x,x,w)\\
		& =0
	\end{align*}
	Finally, we use the Bianchi identity, to show that
	\[
		0=D(u,v,w,t)+D(w,u,v,t)+D(v,w,u,t)=3D(u,v,w,t)
	\]
	\proven
}

\par\noindent
Now, we return to studying the second variation formula. Recall: For a geodesic $\gamma$, and $V$ the variation field of a proper variation of $\gamma$, we have
\[
	E''(0)=-\int_{0}^{a}\iprod{V,\frac{D^{2}}{dt^{2}}V+\mc{R}(V,\dot\gamma)\dot\gamma}\,dt
\]

\thm{
	(Bonnet-Myers V1) If sectional curvature $K$ satisfies $K>\paren{\frac{\pi}{l}}^{2}>0$ for some $l>0$, then no geodesic of length $l$ is minimizing.\nn
	Proof: Let $\gamma$ be a geodesic of length $l$. We need to show the energy of $\gamma$ is not a minimum. Let $\gamma:[0,l]\to{}M$, with $\gamma(0)=p$, $\gamma(l)=q$, and $\norm{\dot\gamma}=1$. Pick $N\in{}T_{p}M$ a unit vector such that $\iprod{N,\dot\gamma(0)}=0$. Let $E(t)$ be the parallel transport of $N$ along $\gamma$, so then $\frac{D}{dt}E\equiv{}0$. Observe that $\forall{}t$, $\iprod{E(t),\dot\gamma(t)}=0$. So if we define
	\[
		V(t)\eqdef\sin\paren{t\frac{\pi}{l}}E(t)
	\]
	then $V(0)=0$ and $V(l)=0$, so $V$ is the variation field of a pinned variation. So now, we can substitute into the second variation formula. We compute
	\[
		\frac{D}{dt}V(t)=\frac{\pi}{l}\cos\paren{t\frac{\pi}{l}}E(t)+0\quad\Rightarrow\quad\frac{D^{2}}{dt^{2}}V(t)=-\paren{\frac{\pi}{l}}^{2}\sin\paren{t\frac{\pi}{l}}E(t)
	\]
	Thus,
	\[
		\iprod{V,\frac{D^{2}}{dt^{2}}V}=-\paren{\frac{\pi}{l}}^{2}\sin^{2}\paren{t\frac{\pi}{l}}{\underbrace{\norm{v}}_{=1}}^{2}
	\]
	Also,
	\[
		\iprod{V,\mc{R}(V,\dot\gamma)\dot\gamma}=\sin^{2}\paren{t\frac{\pi}{l}}\underbrace{\iprod{E,\mc{R}(E,\dot\gamma)\dot\gamma}}_{=K(E,\dot\gamma)}
	\]
	So for a proper variation of $\gamma$ with variation field $V=\restr{f_{s}}_{s=0}$, we have
	\[
		E''(0)=\int_{0}^{l}\underbrace{\sin^{2}\paren{t\frac{\pi}{l}}\vphantom{\paren{\paren{\frac{\pi}{l}}^{2}-K(E,\dot\gamma)}}}_{\ge{}0}\underbrace{\paren{\paren{\frac{\pi}{l}}^{2}-K(E,\dot\gamma)}}_{<0}\,dt
	\]
	So $E''(0)<0$, so for $\abs{s}\ll{}1$, $E(s)<E(0)$, so $\gamma$ is not minimizing.\proven
}
