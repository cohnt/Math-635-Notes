\par\noindent
Recall the theorem and definition stated at the end of the previous lecture...\n

\thm{
	Let $M$ be a Riemannian manifold. Then $\exists\unique\nabla$ on $\mathcal{E}=TM\to{}M$ such that
	\begin{enumerate}[label=(\alph*),leftmargin=4\parindent]
		\item $\nabla$ preserves the Riemannian metric. \i{(This depends on the choice of Riemannian metric.)}
		\item $\forall{}X,Y\in\mf{X}(M)$, $\nabla_{X}Y-\nabla_{Y}X=\brack{X,Y}$. \i{(This does not depend on the choice of Riemannian metric.)}
	\end{enumerate}\up\n
}

\defn{
	This $\nabla$ is called the \u{Levi-Civita connection} on $M$.\n
}

\par\noindent
This theorem is sometimes known as the ``Fundamental Theorem of Riemannian Geometry''. The second condition -- that $\forall{}X,Y\in\mf{X}(M)$, $\nabla_{X}Y-\nabla_{Y}X=\brack{X,Y}$ -- is sometimes called a ``symmetry condition''.\n

\par\noindent
Proof of the theorem: We'll use properties (a) and (b) to find an expression for $\nabla$. Let $X,Y,Z\in\mf{X}(M)$. Then
\begin{align*}
	X\iprod{Y,Z}+Y\iprod{Z,X}-Z\iprod{X,Y} & =\underbrace{\iprod{\nabla_{X}Y,Z}}_{\ptxt{(III)}}+\underbrace{\iprod{Y,\nabla_{X}Z}}_{\ptxt{(I)}}+\underbrace{\iprod{\nabla_{Y}Z,X}}_{\ptxt{(II)}}+\underbrace{\iprod{Z,\nabla_{Y}X}}_{\ptxt{(IV)}}-\underbrace{\iprod{Y,\nabla_{Z}X}}_{\ptxt{(I)}}-\underbrace{\iprod{\nabla_{Z}Y,X}}_{\ptxt{(II)}}\\
	& =\underbrace{\iprod{Y\nabla_{X}Z-\nabla_{Z}X}}_{\ptxt{(I)}}+\underbrace{\iprod{\nabla_{Y}Z-\nabla_{Z}Y,X}}_{\ptxt{(II)}}+\underbrace{\iprod{\nabla_{X}Y,Z}}_{\ptxt{(III)}}+\underbrace{\iprod{\comm{Y,X}+\nabla_{X}Y,Z}}_{\ptxt{(IV)}}\\
	& =\iprod{Y,\comm{X,Z}}+\iprod{X,\comm{Y,Z}}+\iprod{Z,\comm{Y,X}}+2\iprod{Z,\nabla_{X}Y}
\end{align*}
Now, we solve for $2\iprod{\nabla_{X}Y,Z}$:
\[
	2\iprod{\nabla_{X}Y,Z}=X\iprod{Y,Z}+Y\iprod{Z,X}-Z\iprod{X,Y}-\paren{\vphantom{\comm{\tilde{X}}}\iprod{X,\comm{Y,Z}}+\iprod{Y,\comm{X,Z}}+\iprod{Z,\comm{Y,X}}}
\]
This is our defining expression for $\nabla$, since choosing $Z$ in all possible ways defines $\nabla_{X}Y$. We claim that defining $\nabla$ in this way gives us the desired connection. (This part of the proof is tedious, so check it out in the textbook if you're interested.)\proven

\subsection*{Computation of the Christoffel Symbols in Coordinates}

\par\noindent
Let $(x^{1},\ldots,x^{n})$ be coordinates on $U\subseteq{}M$, and define $X_{i}=\pd{}{x^{i}}$, $g_{ij}=\iprod{X_{i},X_{j}}$. Note that $\comm{X_{i},X_{j}}=0$.\n

\par\noindent
Recall: The Christoffel symbols $\Gamma_{ij}^{l}\in{}C^{\infty}(U)$ are defined by $\nabla_{X_{i}}X_{j}=\Gamma_{ij}^{l}X_{l}$. We want to compute $\Gamma_{ij}^{l}$ using the defining expression above. Well,
\[
	2\iprod{\nabla_{X_{i}}X_{j},X_{k}}=2\iprod{\Gamma_{ij}^{l}X_{l},X_{k}}=2\Gamma_{ij}^{l}g_{lk}=X_{i}(g_{jk})+X_{j}(g_{ki})-X_{k}(g_{ij})
\]
Now, we introduce the matrix $g\inv$, the inverse of $(g_{lk})$, with the notation $g\inv=(g^{km})$, so that $g_{lk}g^{km}=\delta_{k}^{m}$. If we multiply both sides by $g\inv$ / $g^{km}$, and sum over $k$, we get
\[
	\underbrace{2\Gamma_{ij}^{l}\underbrace{g_{lk}g^{km}}_{=\delta_{l}^{m}}}_{=2\Gamma_{ij}^{m}}=\sum_{k}g^{km}\paren{\pd{g_{jk}}{x^{i}}+\pd{g_{ki}}{x^{j}}-\pd{g_{ij}}{x^{k}}}
\]
Thus,
\[
	\Gamma_{ij}^{m}=\frac{1}{2}\sum_{k}g^{km}\paren{\pd{g_{jk}}{x^{i}}+\pd{g_{ki}}{x^{j}}-\pd{g_{ij}}{x^{k}}}
\]
\n

\par\noindent
Observe: $\comm{X_{i},X_{j}}=0\Leftrightarrow\Gamma_{ij}^{m}=\Gamma_{ji}^{m}$. So the number of independent indices of the Christoffel symbols is $\frac{n(n+1)}{2}\cdot{}n=\frac{n^{2}(n+1)}{2}$. For example, when $n=2$ (a surface), there are $6$ Christoffel symbols.\n

}

	(HW3) If $M\subset\R^{N}$, with the induced Riemannian metric from the Euclidean metric on $\R^{N}$, and if $\gamma:[a,b]\to{}M$ and $V\in\Gamma_{\gamma}(TM)$, we can define
	\[
		\frac{\bar{D}V}{dt}=\frac{d}{dt}(V:[a,b]\to\R^{n})
	\]
	Claim: If $\frac{D}{dt}$ is the operator associated with the Levi-Civita connection on $M$, then we have $\frac{DV}{dt}(t)=\pi_{\gamma(t)}[\frac{\bar{D}V}{dt}(t)]$, where $\pi_{\gamma(t)}:\R^{N}\to{}T_{\gamma(t)}M$ is the orthogonal projection.\n
}

\subsection*{Geodesics}

\par\noindent
Observe: $TM$ is such that every curve $\gamma$ into $M$ has a natural lift to $TM$.
\[
	\begin{tikzcd}
		& TM \arrow[d]\\
		{[a,b]} \arrow[r,"\gamma"] \arrow[ur,"{(\gamma,\dot\gamma)}"] & M
	\end{tikzcd}
\]
In an abuse of notation, we sometimes write ``$\dot\gamma(t)=\frac{d\gamma}{dt}=(\gamma(t),\dot\gamma(t))$''.\n
We can then consider $\frac{D}{dt}\frac{d\gamma}{dt}$, the acceleration of $\gamma\in\Gamma_{\gamma}(TM)$.\n

\defn{
	$\gamma$ is a \u{geodesic} iff $\frac{D}{dt}(\frac{d\gamma}{dt})=0$.\n
}

\ex{
	Let $M=S^{2}\hookrightarrow\R^{3}$. Then $\gamma(t)=(\cos(t),\sin(t),0)$ is a geodesic, as $(\frac{\bar{D}}{dt}\dot\gamma)(t)=\ddot\gamma(t)=-\gamma(t)$.\n
	$\gamma(t)\perp{}T_{\gamma(t)}S^{2}\Rightarrow\pi_{\gamma(t)}\frac{\bar{D}}{dt}\dot\gamma(t)=0$.\n
}

\par\noindent
Observe:
\begin{enumerate}
	\item If $\gamma$ is a geodesic, then $\frac{d}{dt}\iprod{\frac{d\gamma}{dt}\frac{d\gamma}{dt}}=2\iprod{\frac{D}{dt}\frac{d\gamma}{dt},\frac{D}{dt}\frac{d\gamma}{dt}}=0$, so $\norm{\frac{d\gamma}{dt}}$ is constant with respect to $t$. In other words, the ``speed'' of a geodesic is constant.
	\item If $\gamma$ is a geodesic, $c\in\R$, then $\gamma_{c}(t)\eqdef\gamma(ct)$ is also a geodesic. But other reparameterizations are generally not geodesics. The speed of $\gamma_{c}$ is $\abs{c}$ times the speed of $\gamma$.
\end{enumerate}
