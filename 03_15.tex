\section*{Ricci Curvature}

\par\noindent
For a Riemannian manifold, the full Riemannian curvature at a point $p\in{}M$ is $\mc{R}_{p}\in{}V^{*}\otimes{}V^{*}\otimes{}V^{*}\otimes{}V$, where $V=T_{p}M$. The Ricci Tensor is a partial trace of $\mc{R}$.\n

\defn{
	$\forall{}p\in{}M$, $\forall{}u,v\in{}T_{p}M$, we define the \u{Ricci tensor} by
	\[
		\arraycolsep=2pt
		\Ric_{p}(u,v)=\tr\brack{\begin{array}{rcl}
			T_{p}M & \to & T_{p}M\\
			w & \mapsto & \mc{R}_{p}(u,v)w
		\end{array}}
	\]
}

\par\noindent
Note: The Do Carmo definition of the Ricci tensor has the same convention as ours. It deals with the minus sign by putting $w$ in the second slot of $\mc{R}$.\n

\par\noindent
Note that the trace of an arbitrary linear operator $F:V\to{}V$ can be defined without coordinates as follows. Think of $\Hom(V,V)\cong{}V^{*}\otimes{}V$. The evaluation map $(\alpha,v)\mapsto\alpha(v)$ for $\alpha\in{}V^{*}$, $v\in{}V$ is bilinear, so it corresponds to
\[
	\map{\tr:V^{*}\otimes{}V}{\R}{F}{\tr(F)}
\]
So $\Ric_{p}$ is bilinear.\n

\par\noindent
We want to compute $\Ric_{p}:T_{p}M\times{}T_{p}M\to\R$ in terms of components of $\mc{R}$ with respect to an arbitrary frame. Let $e_{1},\ldots,e_{n}$ be a local frame of $TM$. Then $\mc{R}(e_{i},e_{j})e_{k}=\mc{R}_{ijk}^{l}e_{l}$.\n

\defn{
	\begin{align*}
		\mc{R}_{ij} & \eqdef\Ric(e_{i},e_{j})\\
		& =\tr(w\mapsto\mc{R}(w,e_{i})e_{j})\\
		& =\sum_{k=1}^{n}\paren{\ptxt{$e_{k}$-component of}\;\mc{R}(e_{k},e_{i})e_{j}}\\
		& =\mc{R}_{kij}^{k}\;\ptxt{(with implicit summation)}\\
	\end{align*}
	So $\mc{R}_{ij}=\mc{R}_{kij}^{k}$.\n
}

\par\noindent
(As we will see later, we can interpret this as a sort of averaging.)\n

\par\noindent
Side note: Why do we choose to fix the middle two elements? Well, if you fix the first two or last two, swapping the two unfixed elements flips the sign, so the trace would be identically zero.\n

\par\noindent
Recall that we also have the shorthand notation $\mc{R}_{ijkl}=\iprod{\mc{R}(e_{i},e_{j})e_{k},e_{l}}$.\n

\exer{
	Check that $\mc{R}_{ijkl}=\mc{R}_{ijk}^{a}g_{al}$, i.e., $\mc{R}_{ijk}^{b}=\mc{R}_{ijkl}g^{lb}$ (where $g^{lb}$ is the element of the matrix inverse of $g_{al}$). This implies that $\mc{R}_{ij}=\mc{R}_{kijl}g^{lk}$.\n
}

\par\noindent
Now, let's explore some properties of the Ricci Tensor.\n

\newpage
\prop{
	$\Ric$ is symmetric.\nn
	Proof: Let $e_{1},\ldots,e_{n}$ be an orthonormal frame. Then
	\begin{align*}
		\mc{R}_{ij} & =\sum_{k}\iprod{\mc{R}(e_{k},e_{j})e_{j},e_{k}}\\
		& =\sum_{k}\mc{R}_{kijk}\\
		& =\sum_{k}\mc{R}_{jkki}\\
		& =-\paren{-\sum_{k}\mc{R}_{kjik}}\\
		& =\sum_{k}\mc{R}_{kjik}\\
		& =\mc{R}_{ji}
	\end{align*}
	\proven
}

\defn{
	The \u{Ricci curvature} is
	\[
		\map{\Ric:\set{(p,v)\mid{}p\in{}M,v\in{}T_{p}M,\norm{v}=1}}{\R}{(p,v)}{\Ric_{p}(v,v)\underbrace{\frac{1}{n-1}}_{\mathclap{\ptxt{Do Carmo only}}}}
	\]
}

\subsection*{Geometric Interpretation}

\par\noindent
We'll be using the Do Carmo convention. Let $v$ be a unit tangent vector, and let $(e_{1},\ldots,e_{n})$ be an orthonormal basis of $T_{p}M$, with $e_{n}=v$. Then
\[
	\Ric(v)=\frac{1}{n-1}\sum_{i=1}^{n-1}K(e_{i},v)
\]
So we're effectively averaging the sectional curvature of the planes produced by $v$ with each of the basis vectors $e_{1},\ldots,e_{n-1}$. This is true because
\[
	\Ric(v,v)=\sum_{i=1}^{n}\underbrace{\iprod{\mc{R}(e_{i},v)v,e_{i}}}_{\hspace{35pt}\mathclap{=\left\{\begin{array}{ll}
			K(e_{i},v) & i<n\\
			0 &
		\end{array}\right.}}
\]

\defn{
	The \u{scalar curvature} $S:M\to\R$ is defined by $\forall{}p\in{}M$, $S(p)=\tr(\Ric_{p})$.\n
}

\thm{
	(Bonet-Myers V2) Suppose $(M,g)$ is a Riemannian manifold. If $\Ric>\paren{\frac{\pi}{l}}^{2}$ (using the function form of $\Ric$, not the tensor), then no geodesic of length $l$ is minimizing.\nn
	Proof: Pick $(E_{1}(0),\ldots,E_{n-1}(0))$, an orthonormal basis of $\vspan(\dot\gamma(0))^{\perp}$. Then parallel transport the $E_{j}$'s along $\gamma$ to get $E_{j}(t)$, a parallel frame (i.e. $\frac{D}{dt}E_{j}\equiv{}0$) that is orthonormal at every $t$, and orthogonal to $\dot\gamma(t)$. Now, $\forall{}j$, let $V_{j}(t)=\sin(\pi{}t)E_{j}(t)$. We consider $n-1$ proper variations of $\gamma$, and $\forall{}j=1,\ldots,n-1$, we have, as before,
	\[
		E_{j}''(0)\underbrace{=\cdots=}_{\mathclap{\substack{|\\ \ptxt{Same computation as Bonet-Myers V1}}}}\int_{0}^{l}\sin^{2}\paren{t\frac{\pi}{l}}\paren{\paren{\frac{\pi}{l}}^{2}-K(E_{j},\dot\gamma)}\,dt
	\]
	Now, we average:
	\[
		\frac{1}{n-1}\sum_{j=1}^{n-1}E_{j}''(0)=\int_{0}^{l}\sin^{2}(t\frac{\pi}{l})\paren{\paren{\frac{\pi}{l}}^{2}-\Ric_{\gamma(t)}(\dot\gamma)}\,dt
	\]
	This quantity is less than $0$ by our assumption, so $\exists{}j$ such that $E_{j}''(0)<0$, so the geodesic isn't minimizing.\proven
}

\section*{Jacobi Fields}

\par\noindent
Our motivation comes from the second variation formula: it involves the operator $\frac{D^{2}}{dt^{2}}V+\mc{R}(V,\dot\gamma)\dot\gamma$. We're interested in studying fields where this is equal to $0$.\n

\defn{
	Let $\gamma$ be a geodesic. Then $V\in\Gamma_{\gamma}(TM)$ is a \u{Jacobi field} iff
	\[
		\frac{D^{2}}{dt^{2}}V+\mc{R}(V,\dot\gamma)\dot\gamma=0
	\]
}

\par\noindent
Note: We don't require $V$ to vanish at the endpoints.\n

\prop{
	Let $f:(-\varepsilon,\varepsilon)\times[0,a]\to{}M$ be a smooth variation of $\gamma$ by geodesics, i.e., $\forall{}s$, $t\mapsto{}f(s,t)$ is a geodesic. Then the variation field $V=\restr{\d_{s}f}_{s=0}$ is a Jacobi field.\nn
	Proof: Our assumption of $f$ is true iff $\frac{D}{dt}\d_{t}f=0$, so
	\[
		0=\frac{D}{ds}\frac{D}{dt}\d_{t}f=\frac{D}{dt}\frac{D}{ds}\d_{t}f+\mc{R}(\d_{s}f,\d_{t}f)\d_{t}f=\frac{D^{2}}{dt^{2}}\d_{s}f+\mc{R}(\d_{s}f,\d_{t}f)\d_{t}f
	\]
	If we restrict to $s=0$, we get the Jacobi operator.\proven
}

	(HW) Prove that the converse is true as well.\n
