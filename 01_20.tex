\par\noindent
The overall goal of this course is to build a notion of distance on a manifold. But in order to have such a notion, we need additional structure. Specifically, we need a Riemannian metric: a $C^{\infty}$ assignment of an inner product to each tangent space $T_{p}M$, $\forall{}p\in{}M$.\n

\ex{
	Say $S\subset\R^{N}$ is a submanifold. $\forall{}p\in{}S$, $T_{p}S\hookrightarrow\R^{N}$, so it inherits the Euclidean inner product of $\R^{N}$. We can use this to define the length of a curve $\gamma:(a,b)\to{}S$ -- $\int_{a}^{b}\norm{\dot\gamma(s)}\,ds$, where $\norm{\dot\gamma}=\sqrt{\iprod{\dot\gamma,\dot\gamma}}$.\n
}

\par\noindent
For a systematic study of these ideas, we need to review:
\begin{itemize}
	\item Vector bundles
	\item Connections on vector bundles
\end{itemize}

\par\noindent
The basic idea of a connection is that a Riemannian metric allows us to talk about parallel transport of vectors along curves. Specifically, a connection allows us to differentiate sections of a vector bundle along curves in the base.

\subsection*{Vector Bundles (Review)}

\defn{
	A \u{vector bundle} (of rank $r$) over $M$ is a surjective submersion $\pi:\mathcal{E}\to{}M$ \st{}
	\begin{enumerate}[label=\alph*), itemsep=0pt, topsep=0pt, leftmargin=4\parindent]
		\item For each $p\in{}M$, the fiber $\mathcal{E}_{p}\eqdef\pi\inv(p)$ is a vector space (of rank $r$).
		\item There's an (open\footnote{In the future, assume that any covering is open, unless explicitly stated otherwise.}) covering $\set{U_{\alpha}}$ of $M$, and diffeomorphisms $\chi_{\alpha}:\pi\inv(U_{\alpha})\to{}U_{\alpha}\times\R^{r}$ called \u{local trivializations}, such that
		\begin{enumerate}[label=\roman*)]
			\item $\forall\alpha$, the following diagram commutes
			\[
				\begin{tikzcd}[ampersand replacement=\&]
					\pi\inv(U_{\alpha}) \arrow[rr, "\chi_{\alpha}"] \arrow[dr,"\pi"] \&[-10pt] \&[-10pt] U_{\alpha}\times\R^{r} \arrow[dl]\\
					\& U_{\alpha}
				\end{tikzcd}
			\]
			\item $\forall{}p\in{}U$, $\restr{\chi_{\alpha}}{\mathcal{E}_{p}}:\mathcal{E}_{p}\to\set{p}\times\R^{r}\cong\R^{r}$ is a linear isomorphism.
		\end{enumerate}
	\end{enumerate}
	\up\n
}

\par\noindent
Colloquially, we think of $\mathcal{E}$ as a family $\set{\mathcal{E}_{p}}$ of vector spaces, parameterized by $M$.\n

\ex{
	$TM$ and $T^{*}M$ are examples of vector bundles.\n
}

\defn{
	Given a vector bundle $\pi:\mathcal{E}\to{}M$, let $\Gamma(\mathcal{E})\eqdef\set{s:M\to\mathcal{E}\mid{}s\ptxt{ smooth and }\pi\of{}s\equiv{}I_{M}}$, the set of mooth sections of $\mathcal{E}$.\n
}

\defn{
	The \u{zero section} is $s$ \st{} $\forall{}p\in{}M$, $s(p)=0\in\mathcal{E}_{p}$.\n
}

\ex{
	If $\mathcal{E}=TM$, then $\mathcal{E}_{p}=T_{p}M$, so
	\begin{itemize}[topsep=0pt, itemsep=0pt, leftmargin=3\parindent]
		\item $\Gamma(TM)=\mathfrak{X}(M)$, the set of all smooth vector fields on $M$.
		\item $\Gamma(T^{*}M)$ is the set of all smooth $1$-forms on $M$.
	\end{itemize}
	\up\n
}

\par\noindent
Observe: $\Gamma(\mathcal{E})$ is a module over $C^{\infty}(M)$ -- $\forall{}s,t\in\Gamma(\mathcal{E}),f\in{}C^{\infty}(M)$, $(fs+t)(p)=f(p)s(p)+t(p)$.\n

\defn{
	A \u{moving frame} is a collection of sections $(E_{1},\ldots,E_{r})$ over $U$ \st{} $\forall{}p\in{}U$, $(E_{1}(p),\ldots,E_{r}(p))$ is a basis of $\mathcal{E}_{p}$.\n
}

\par\noindent
Observe: Using our notion of modules, a local trivialization $\chi:\pi\inv(U)\to{}U\times\R^{r}$ defines a moving frame on $U$. If $(x^{1},\ldots,x^{n})$ are coordinates on $U\subseteq{}M$, then $\set{\pd{}{x^{1}},\ldots,\pd{}{x^{n}}}$ is a moving frame of $TM$ on $U$.\n

\par\noindent
Given $\chi$, define $E_{j}(p)\in\mathcal{E}_{p}$ such that $\chi(p,E_{j}(p))=(p,\ang{0,\ldots,1,\ldots,0})$ (with a $1$ in the $j$th entry). Then we have
\[
	\begin{tikzcd}
		\pi\inv(U) \arrow[rr, shift left=1, "\chi"] \arrow[dr,"\pi", swap] &[-10pt] &[-10pt] U\times\R^{r} \arrow[dl] \arrow[ll, shift left=1,"\chi\inv"]\\
		& U \arrow[ur, bend right=30, "p\mapsto(p{,}\ang{0,\ldots,1,\ldots,0})", swap]
	\end{tikzcd}
\]
Then $\forall{}s\in\Gamma(\restr{\mathcal{E}}{U})$, $\exists{}f_{j}:U\to\R$ such that $\forall{}p\in{}U$,
\[
	s(p)=\sum_{j=1}^{r}f_{j}(p)E_{j}(p)
\]
This is true for any section $s:U\to\pi\inv(U)=\restr{\mathcal{E}}{U}$, and, in fact, these $f_{j}$'s are unique, simply because $s(p)\in\mathcal{E}_{p}$, and $\set{E_{j}(p)}$ is a basis of $\mathcal{E}_{p}$.\n

\lemma{
	$s$ is smooth iff $\forall{}j$, $f_{j}$ is smooth.\n
}

\defn{
	The \u{trivial bundle} of rank $r$ is $\pi:M\times\R^{r}\to{}M$.\n
}

\subsection*{Operations with Vector Bundles}

\par\noindent
Principle: Any operation or construction that one can do with vector spaces, that is natural with respect to linear isomorphisms, can be extended to vector bundles, by doing such an operation or construction fiber-wise.\n

\ex{
	\begin{itemize}[topsep=0pt]
		\item $V\rightsquigarrow{}V^{*}$
		\item $V,W\rightsquigarrow{}V\oplus{}W$
		\item $V,W\rightsquigarrow{}V\otimes{}W$
		\item $V,W\rightsquigarrow{}\Hom(V,W)$
		\item $V\subset{}w\rightsquigarrow{}W/V$
	\end{itemize}
	\up\n
}

\par\noindent
Idea: Given any two bundles $\mathcal{E},\mathcal{F}\to{}M$ over $M$, we can form the ``Whitney direct sum'', $\mathcal{E}\oplus\mathcal{F}\to{}M$, such that $\forall{}p\in{}M$, $(\mathcal{E}\oplus\mathcal{F})_{p}=\mathcal{E}_{p}\oplus\mathcal{F}_{p}$. This already defines what $\mathcal{E}\oplus\mathcal{F}$ has to be as a set:
\[
	\mathcal{E}\oplus\mathcal{F}=\bigsqcup_{p\in{}M}\mathcal{E}_{p}\oplus\mathcal{F}_{p}
\]
The question is: how do we give such a set the topology and $C^{\infty}$ structure of a vector bundle?
