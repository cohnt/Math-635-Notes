\par\noindent
Recall: A Riemannian metric $g$ on $M$ is a covariant $2$-tensor (i.e. a tensor of type $(0,2)$), such that, at each point, $g_{p}$ is symmetric and positive definite.\n

\prop{
	Any $C^{\infty}$ manifold $M$ has infinitely many Riemannian metrics.\nn
	Proof 1: We can embed $M\hookrightarrow\R^{N}$ (for some $N$), and use the induced metric.\proven\up\nn
	Proof 2: We can use a partition of unity. Let $\set{U_{\alpha},\phi_{\alpha}}$ be an atlas of $M$, and $\set{\chi_{\alpha}}$ a subordinate partition of unity. (Recall: A subordinate partition of unity means $\forall\alpha$, $\chi_{\alpha}\in{}C^{\infty}_{0}(M)$, $\set{\supp\chi_{\alpha}}$ is a locally finite cover of $M$, and $\forall{}p\in{}M$, $\sum_{\alpha}\chi_{\alpha}(p)=1$.) $\forall\alpha$, let $g^{\alpha}$ be the Riemannian metric on $U_{\alpha}$ \st{} $\phi_{\alpha}$ is an isometry (to the standard flat metric on $\R^{n}$). That is, $(g^{\alpha})_{ij}=\delta_{ij}$. Define, $\forall{}p\in{}M$, $\forall{}u,v\in{}T_{p}M$,
	\[
		g_{p}(u,v)=\sum_{\alpha}\underbrace{\chi_{\alpha}(p)g_{p}^{\alpha}(u,v)}_{\hspace{-60pt}\mathrlap{\ptxt{Interpret this as $0$ if $p\not\in{}U_{\alpha}$.}}}
	\]
	Basically, we're extending $\chi_{\alpha}g^{\alpha}$ to $0$ outside $U_{\alpha}$. Now that we've define $g$, we need to check that it satisfies the appropriate properties. Symmetric matrices form a closed group under addition, so clearly, every $g_{p}$ is symmetric. And because $\chi_{\alpha}\ge{}0$, we can also assert that $g_{p}$ is positive semidefinite. Finally we have to check that\break $g_{p}(u,u)=0\Rightarrow{}u=0$. Well, $\exists\alpha$ \st{} $\chi_{\alpha}(p)>0$, so $p\in{}U_{\alpha}$. Thus, we must have $g_{p}^{\alpha}(u,u)=0$, so we must have $u=0$ by the positive definiteness of $g_{p}^{\alpha}$.\proven
}

\par\noindent
Observe: An analogous proof shows that any vector bundle has infinitely many $C^{\infty}$ sections, where we use local trivializations instead of coordinates.\n

\par\noindent
Note that positive-definiteness is important! It's not true that $M$ has a non-degenerate symmetric covariant $2$-tensor of any arbitrary signature. For example, general relativity cares about the signature $(-,+,+,+)$, and such a tensor (with all relevant properties) doesn't always exist.\n

\defn{
	Let $G$ be a Lie group. (Recall that this means $G$ is a manifold, and a group, with the group operation and inversion $C^{\infty}$ functions with respect to the smooth structure.) $\forall{}k\in{}G$, we have the left translation map $L_{k}:G\to{}G$, where $L_{k}(h)=kh$. A Riemannian metric on $G$ is said to be \u{left-invariant} iff $\forall{}k\in{}G$, $L_{k}$ is an isometry.\n
}

\prop{
	Let $\mathfrak{g}=T_{e}G$, the Lie algebra (where $e\in{}G$ is the identity element of the group). Then given any positive definite inner product $\giprod_{e}$ on $\mathfrak{g}$, there exists a unique left-invariant Riemannian metric $g$ on $G$ \st{} $g_{e}=\giprod_{e}$.\nn
	Proof: $\forall{}k\in{}G$, we require (by left-invariantness) that $d(L_{k})_{e}:(\mathfrak{g},\giprod_{e})\to(T_{k}G,g_{k})$ is an isometry. We define $g$ this way -- $g_{k}$ is the ``pushforward'' of $\giprod_{e}$ by $d(L_{k})_{e}$. Specifically, $\forall{}u,v\in{}T_{k}G$,
	\[
		g_{k}(u,v)=\iprod{d(L_{k}\inv)(u),d(L_{k}\inv)(v)}_{e}=\iprod{d(L_{-k})(u),d(L_{-k})(v)}_{e}
	\]
	We claim that this is smooth, using the smoothness of the map $\map{G\times{}G}{G}{(k,h)}{kh}$.\n
}

\par\noindent
Question: We can also define everything in terms of right-invariant metrics. Do bi-invariant metrics exist? Answer (which we will show in HW): It depends on $G$!\n

\par\noindent
If $M$ and $N$ are Riemannian manifolds, then $M\times{}N$ has a natural Riemannian metric. $\forall{}(p,q)\in{}M\times{}N$, $T_{(p,q)}(M\times{}N)=(T_{p}M)\oplus(T_{q}N)$. The product metric is defined as follows: Declare $T_{p}M$ and $T_{q}N$ to be orthogonal. With $g^{M},g^{N}$ the Riemannian metrics for $M,N$ (respectively), we have the product metric on $M\times{}N$ (as a block matrix):
\[
	g^{M\times{}N}=\paren{\begin{array}{c|c}
		g^{M}\vphantom{g^{M^{M}}} & 0\\ \hline
		0 & g^{N}\vphantom{g^{N^{N}}}
	\end{array}}
\]

\defn{
	Let $M,N$ be Riemannian manifolds. A smooth map $F:M\to{}N$ is a \u{local isometry} iff $\forall{}p\in{}M$,
	\[
		dF_{p}:(T_{p}M,g^{M}_{p})\to(T_{F(p)}N,g^{N}_{F(p)})
	\]
	is a linear isometry.\n
}

\par\noindent
Note that local isometries are local diffeomorphisms, by the inverse function theorem.\n

\defn{
	A surjective $C^{\infty}$ map $\pi:M\to{}N$ is a \u{smooth covering map} iff $\forall{}q\in{}N$, there's a neighborhood $V$ of $q$ \st{} $\pi$ maps the connected components of $\pi\inv(U)$ diffeomorphically onto $V$. That is,
	\[
		\pi\inv(U)=\bigsqcup_{j}U_{j},\qquad\forall{}j,\restr{\pi}{U_{j}}:U_{j}\overset{\sim}{\to}V
	\]
}

\defn{
	Let $\pi$ be a smooth covering map. The \u{automorphism group} is $\Aut(\pi)=\set{F:M\to{}N\mid{}\pi=F\of\pi}$. We say that such an $F$ ``shuffles the fibers''. $\Aut(\pi)$ is a group under composition.\n
}

\defn{
	$\pi$ is \u{normal} iff $\Aut(\pi)$ acts transitively on the fibers.\n
}

\ex{
	$S^{n}\overset{\pi}{\to}\RP^{n}$. $\Aut(\pi)=\set{\ptxt{identity map, antipodal map}}\cong\Z_{2}$.\n
	$\R^{n}\overset{\pi}{\to}\R^{n}/\Z^{n}$.\n
}

\ex{
	(HW) $\mathcal{H}=\set{(x,y)\in\R^{2}\mid{}y>0}$. For any genus $g$ surface $G$, with $g>1$, $\exists\pi:\mathcal{H}\to{}G$.\n
}

\thm{
	Let $\pi:M\to{}N$ be a smooth normal covering map. Let $g^{M}$ be a Riemannian metric on $M$ \st{} $\forall{}F\in\Aut(\pi)$ is an isometry. Then there's a unique Riemannian metric $g_{N}$ on $N$ \st{} $\pi$ is a local isometry.\nn
	Proof: Let $q\in{}N$, and $V$ a neighborhood of $q$. Then $\pi\inv(V)=\bigsqcup_{j}U_{j}$ (where the $U_{j}$ are the connected components) such that $\restr{\pi}{U_{j}}:U_{j}\overset{\sim}{\to}V$. We want to choose a metric on $V$ \st{} $\forall{}j$, $\restr{\pi}{U_{j}}$ is an isometry. Well, choose a $j_{0}$, and put such a metric on $V$. $\forall{}j\ne{}j_{0}$, $\exists{}F\in\Aut(\pi)$ \st{} $F(U_{j})=F(U_{j_{0}})$. $F$ is an isometry, by assumption, so $\pi$ is an isometry on $U_{j}$.\proven
}
