\section*{The Variantional Point of View of Geodesics}

\par\noindent
This material is covered in Do Carmo, chapter 9 \sectionSymbol{}2, and in parts of Lee Riemannian Manifolds, chapters 6 and 10.\n

\par\noindent
Throughout these notes, let $M$ be a Riemannian manifold, $p,q\in{}M$, and $a>0$.\n

\defn{
	$\Omega_{pq}^{a}=\set{c:[0,a]\to{}M\mid{}c\;\ptxt{is continuous and piecewise $C^{1}$},c(0)=p,c(a)=q}$. In this case, piecewise $C^{1}$ means $\forall{}c$, there's a partition $0=t_{0}<t_{1}<\cdots<t_{N}=a$ such that $\forall{}i$, $\restr{c}_{[t_{i},t_{i+1}]}$ is $C^{1}$. In other words, $c$ is $C^{1}$ on $(t_{i},t_{i+1})$, and the one sided limits
	\[
		\lim_{t\to{}t_{i}^{-}}\frac{dc}{dt}\quad\lim_{t\to{}t_{i+1}^{+}}\frac{dc}{dt}
	\]
	exist.\n
}

\defn{
	We define two functionals on this space.
	\begin{enumerate}[topsep=0pt, itemsep=0pt, leftmargin=4\parindent]
		\item From Lee: The length functional $\map{L:\Omega_{pq}^{a}}{\R}{c}{L(c)=\int_{0}^{a}\norm{\frac{dc}{dt}}\,dt}$\n
		Any minima of $L$, if one exists, corresponds to shortest paths between $p$ and $q$.
		\item From Do Carmo: The energy functional $\map{E:\Omega_{pq}^{a}}{\R}{c}{E(c)=\frac{1}{2}\int_{0}^{a}\norm{\frac{dc}{dt}}^{2}\,dt}$
	\end{enumerate}\up\n
}

\lemma{
	$\forall{}c\in\Omega_{pq}^{a}$, $L(c)^{2}\le{}2a{}E(c)$, with equality iff $\norm{\dot{c}}$ is constant.\nn
	Proof: Use the Cauchy-Shwarz inequality for functions on $[0,a]$: If $f,g:[0,1]\to\R$, then
	\[
		\paren{\int_{0}^{a}fg\,dt}^{2}\le\paren{\int_{0}^{a}f^{2}\,dt}\paren{\int_{0}^{a}g^{2}\,dt}
	\]
	Given a path $c$, apply Cauchy-Schwarz to $f=\norm{\frac{dc}{dt}}$, with $g\equiv{}1$.\proven
}

\cor{
	Suppose $\gamma\in\Omega_{pq}^{a}$ is a minimizing geodesic. Then $\forall{}c\in\Omega_{pq}^{a}$, $E(\gamma)\le{}E(c)$, with equality iff $c$ is a minimizing geodesic.\nn
	Proof: $\norm{\dot\gamma}$ is constant, so $E(\gamma)=\frac{1}{2a}L(\gamma)^{2}\le\frac{1}{2a}L(c)^{2}\le{}E(c)$. Thus, if $E(\gamma)=E(c)$, then everything must be equal, so $L(\gamma)=L(c)$, so $c$ is a minimizing path, so $c$ is a geodesic.\proven
}

\par\noindent
We want to look for minimizers of $E$. This is a hard problem, and in fact, the may not exist.\n

\par\noindent
The idea of the calculus of variations is to differentiate $E$, and then look for critical points. This is crazy, because $\Omega_{pq}^{a}$ is not a manifold. But one can still define variations of $\gamma\in\Omega_{pq}^{a}$. Colloquially, these are smooth paths in $\Omega_{pq}^{a}$ that pass through $\gamma$.\n

\defn{
	Let $\gamma\in\Omega_{pq}^{a}$. A \u{proper variation} (or \u{pinned variation}) of $\gamma$ is
	\[
		f:\underset{s}{(-\varepsilon,\varepsilon)}\times\underset{t}{[0,a]}\to{}M
	\]
	such that
	\begin{enumerate}[itemsep=0pt, topsep=0pt, leftmargin=4\parindent, label=(\roman*)]
		\item $f(0,t)=\gamma(t)$
		\item There is a partition $0=t_{0}<t_{1}<\cdots<t_{N}=a$ such that $\forall{}i$, $\restr{f}_{(-\varepsilon,\varepsilon)\times[t_{i},t_{i+1}]}$ is $C^{1}$.
		\item $\forall{}s$, $f(s,0)=p$ and $f(s,a)=q$. That is, $\forall{}s\in(-\varepsilon,\varepsilon)$, $(t\mapsto{}f(s,t))\in\Omega_{pq}^{a}$. In this sense, $s\mapsto(t\mapsto{}f(s,t))$ is a ``smooth'' curve in $\Omega_{pq}^{a}$, passing through $\gamma$ at $t=0$.
	\end{enumerate}\up\n
	Conditions (i) and (ii) are what make it a variation; condition (iii) makes it proper/pinned.\n
}
\newpage

\defn{
	For $f$ a proper variation of $\gamma$, $\forall{}t$, $V(t)=\d_{s}f(0,t)\in{}T_{\gamma(t)}M$ is the \u{variation field} of the variation. So $V\in\Gamma_{\gamma}(TM)$.\n
}

\par\noindent
Observe: Because the variation is proper, we must have $V(0)=0$.\n

\lemma{
	Given any $V\in\Gamma_{\gamma}(TM)$ \st{} $V(0)=0$, $V(a)=0$, there exists a proper variation $f$ whose variation field is $V$.\nn
	Proof: Let $f(s,t)=\exp_{\gamma(t)}(sV(t))$. We need $\abs{s}<\varepsilon$ to be nonzero, but because of the compactness of the curve, we can construct a finite subcover, so that $\abs{s}>0$.\proven
}

\par\noindent
Idea: Differentiate the energy $E$ w.r.t. a given varation of $\gamma\in\Omega_{pq}^{a}$.\n
Computation: Let $f$ as above, a proper variation of $\gamma\in\Omega_{pq}^{a}$. Define
\[
	E(s_{0})=\frac{1}{2}\int_{0}^{a}\norm{\d_{t}f(s,t)}^{2}\,dt=E(\restr{f}_{s=s_{0}})
\]
Then compute
\begin{align*}
	\frac{dE}{ds} & =\frac{1}{2}\int_{0}^{a}\frac{d}{ds}\iprod{\d_{t}f,\d_{t}f}\,dt\\
	& =\cancel{\frac{1}{2}}\int_{0}^{a}\cancel{2}\iprod{\frac{D}{ds}\d_{t}f,\d_{t}f}\,dt\\
	& =\int_{0}^{a}\iprod{\frac{D}{dt}\d_{s}f,\d_{t}f}\,dt\\
	\ptxt{(integration by parts)}\quad & =\brack{\iprod{\d_{s}f,\d_{t}f}}_{t=0}^{t=a}-\int_{0}^{a}\iprod{\d_{s}f,\frac{D}{dt},\d_{t}f}\,dt
\end{align*}
Where the integration by parts succeeds because
\[
	\frac{d}{dt}\iprod{\d_{s}f,\d_{t}f}=\iprod{\frac{D}{dt}\d_{s}f,\d_{t}f}+\iprod{\d_{s}f,\frac{D}{dt}\d_{t}f}
\]
Note that we have to adjust the term outside of the integral due to corners, but the boundary terms will all appear in a similar form. We conclude with
\[
	\frac{d}{ds}E(0)=-\int_{0}^{a}\left<\vphantom{\frac{D}{dt}}\right.\underbrace{V(t)\vphantom{\frac{D}{dt}}}_{\restr{\d_{s}f}_{s=0}},\frac{D}{dt}\frac{d\gamma}{dt}\left.\vphantom{\frac{D}{dt}}\right>\,dt-\underbrace{\sum_{i}\left<\vphantom{\dot\gamma}\right.V(t_{i}),\underbrace{\Delta\dot\gamma(t_{i})}_{\mathclap{=\frac{d\gamma}{dt}(t_{i}^{+})-\frac{d\gamma}{dt}(t_{i}^{-})}}\left.\vphantom{\dot\gamma}\right>}_{\ptxt{corner terms}}
\]
which is the first variation formula.\n

\par\noindent
Now, choose
\[
	V(t)=\left\{\begin{array}{ll}
		\frac{D}{dt}\frac{d\gamma}{dt} & t\in(t_{i},t_{i+1})\;\ptxt{for some}\;i\\
		V(t_{i}) & t=t_{i}\;\ptxt{for some}\;i
	\end{array}\right.
\]
We conclude that if $\gamma\in\Omega_{pq}^{a}$ is a critical point of $E$ in the sense that for all variations $f$, $E'(0)=0$, then $\gamma$ is a geodesic!\n
