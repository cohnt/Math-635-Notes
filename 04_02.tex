\par\noindent
Recall our definitions: $\bar{M}$ is a Riemannian manifold, $M\subseteq\bar{M}$ is a submanifold with the induced metric. $\forall{}X,Y\in\mf{X}(M)$, $B(X,Y)=\bar\nabla_{\bar{X}}\bar{Y}-\nabla_{X}Y\in(T_{p}M)^{\perp}$, where $\bar{X}$ and $\bar{Y}$ are extensions of $X$ and $Y$ (respectively) to $\bar{M}$. (This is normal to $M$.) We can also think of it as $\bar\nabla_{\bar{X}}\bar{Y}=\nabla_{X}Y+B(X,Y)$.

\begin{itemize}
	\item $B$ is a symmetric tensor -- $\forall{}p\in{}M$, $B_{p}:T_{p}M\times{}T_{p}M\to(T_{p}M)^{\perp}\subseteq{}T_{p}\bar{M}$.
	\item If we define, $\forall{}p\in{}M$, $\nu_{p}\in(T_{p}M)^{\perp}$ a unit normal vector, then $S_{\nu}:T_{p}M\to{}T_{p}M$ is the shape operator, with the defining property $\forall{}x,y\in{}T_{p}M$, $\iprod{S_{\nu}(x),y}=\iprod{B(x,y),\nu}$.
\end{itemize}

\par\noindent
Because $B$ is symmetric, $S_{\nu}$ is self-adjoint, so there's an orthonormal eigenbasis $(e_{1},\ldots,e_{n})$ of $T_{p}M$ and $\kappa_{1},\ldots,\kappa_{n}$ such that $S_{\nu}(e_{i})=\kappa_{i}e_{i}$. The $e_{i}$ are the principal directions, and the $\kappa_{i}$ are the principal curvatures.\n

\par\noindent
Observe: If $\codim{}M=1$, then $\delta_{ij}\kappa_{i}=\iprod{S_{\nu}(e_{i}),e_{j}}=\iprod{B(e_{i},e_{j}),\nu}$ by orthonormality. This is just the $\nu$-component of $B(e_{i},e_{j})$, so $B(e_{i},e_{j})=\delta_{ij}\kappa_{i}\nu$.\n

\thm{
	(Weingarten Formula) Pick $p\in{}M$ and $\nu\in(T_{p}M)^{\perp}$. Extend $\nu$ to a unit normal vector field $N$ on $M$.\n Then $\forall{}x\in{}T_{p}M$, the Weingarten Formula says
	\[
		S_{\nu}(x)=-(\bar\nabla_{x}N)(p)
	\]
	(This is computed via a curve in $M$ with velocity $x$.)\nn
	Proof: Let $x,y\in{}T_{p}M$, and extend these to vector fields $\bar{X}$ and $\bar{Y}$ tangent to $M$. Then
	\[
		\iprod{S_{\nu}(x),y}=\iprod{B(x,y),N_{p}}(p)=\iprod{\bar\nabla_{\bar{X}}\bar{Y}-\nabla_{X}Y,N}(p)=\iprod{\bar\nabla_{\bar{X}}\bar{Y},N}(p)
	\]
	because $\iprod{\nabla_{X}Y,N}(p)=0$, as $(\nabla_{X}Y)_{p}\in{}T_{p}M$ and $N\in(T_{p}M)^{\perp}$ are orthogonal. On the other hand, $0=\restr{\bar{Y},N}_{M}$, so
	\[
		0=\bar{X}\iprod{\bar{Y},N}=\iprod{\bar\nabla_{\bar{X}}\bar{Y},N}+\iprod{\bar{Y},\bar\nabla_{\bar{X}}N}
	\]
	The first term in the last expression is $\iprod{S_{\nu}(x),y}$ by the above. So we have
	\[
		\iprod{S_{\nu}(x),y}=-\iprod{\bar{Y},\bar\nabla_{\bar{X}}N}=-\iprod{y,\bar\nabla_{x}N}=\iprod{-\bar\nabla_{x}N,y}\qquad\Rightarrow\qquad{}S_{\nu}(x)=-(\bar\nabla_{x}N)(p)
	\]
	\proven
}

\par\noindent
Consider the case where $\bar{M}=\R^{n+1}$, and $\dim{}M=n$. We identify tangent spaces with subspaces. Let $U$ be the domain of $N$ above. Then $N$ is a map $N:U\to\S^{n}$, the unit sphere. We have
\[
	{\arraycolsep=2pt
	\begin{array}{rcl}
		T_{p}M & \cong & T_{N(p)}\S^{n}\\
		S_{N_{p}}:T_{p}M\to{}T_{p}M & \cong & T_{N(p)}\S^{n}
	\end{array}}
\]
Claim: $S_{N_{p}}:T_{p}M\to{}T_{N(p)}\S^{n}$ is equal to $-dN_{p}$.\n
Check this: Let $c(t)$ be a curve on $M$ with $p=c(0)$, $x=\dot{c}(0)$. Then
\[
	S_{N_{p}}(x)=-\bar\nabla_{x}N=\restr{-\frac{d}{dt}N_{x(t)}}_{t=0}=dN_{p}(x)
\]

\newpage
\ex{
	\begin{enumerate}[label=\alph*)]
		\item Let $M=\S^{n}_{R}$, the sphere of radius $R$ in $\R^{n+1}$. Let $N$ be the outward pointing unit normal.
		\[
			\map{N:\S^{n}_{R}}{\S^{n}}{p}{\frac{1}{R}p}\qquad\Rightarrow\qquad{}S_{N}=\frac{1}{R}\Id
		\]
		In this case, every direction is principal, with corresponding principal curvature $\kappa=-\frac{1}{R}$.
		\item Let $M$ be a cylinder of radius $R$ in $\R^{3}$. Specifically, with parameterization
		\[
			\vec{r}(\theta,z)=(R\cos\theta,R\sin\theta,z)
		\]
		Let $N(\theta,z)=(\cos\theta,\sin\theta,0)$. We have the basis of $T_{p}M$:
		\[
			\left\{\begin{array}{l}
				\pd{}{\theta}=(-R\sin\theta,R\cos\theta,0)\\
				\pd{}{z}=(0,0,1)
			\end{array}\right.
		\]
		Then we have $T_{p}M\cong{}T_{N_{p}}\S^{2}$. The matrix of $dN$ with respect to the above basis is
		\[
			\begin{pmatrix}
				-\frac{1}{R} & 0\\
				0 & 0
			\end{pmatrix}
		\]
		So we conclude that $\pd{}{\theta}$ is a principal direction, with corresponding principal curvature $\kappa_{1}=-\frac{1}{R}$. $\pd{}{z}$ is the other principal direction, with corresponding principal curvature $\kappa_{2}=0$.
	\end{enumerate}
}

\cor{
	If $x,y\in{}T_{p}M$ are orthonormal, then
	\[
		\bar{K}(x,y)=K(x,y)+\norm{B(x,y)}^{2}-\iprod{B(x,x),B(y,y)}
	\]
	Proof: Simply use $\bar{K}(x,y)=R(x,y,y,x)$.\proven
