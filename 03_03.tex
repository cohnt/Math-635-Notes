\cor{
	Let $M$ be a connected Riemannian manifold. Then $\forall{}p,q\in{}M$,
	\[
		d(p,q)\eqdef\inf\set{l(c)\mid{}\ptxt{$c$ continuous and piecewise $C^{1}$ curve from $p$ to $q$}}
	\]
	is a distance function on $M$.\nn
	Proof: All that remains to be proved is $d(p,q)=0\Rightarrow{}p=q$. So by contraposition, it's enough to show $p\ne{}q\Rightarrow{}d(p,q)>0$. Well, if $p\ne{}q$, then $\exists\varepsilon>0$ \st{} $\restr{\exp_{p}}_{B_{\varepsilon}(0)}$ is a diffeomorphism onto its image, and $q\not\in\exp_{p}(B_{\varepsilon}(0))$. Let $c$ be a path from $p$ to $q$. It's enough to show $l(c)>\varepsilon>2$.\nn
	Well, by continuity, $\exists{}t_{1}>0$ \st{} $c(t_{1})\in{}S_{\varepsilon/2}$, the geodesic sphere of radius $\varepsilon/2$ centered at $p$. But $d(p,c(t_{1}))$ is the length of any corresponding radial geodesic, which is $\varepsilon/2$. So $l(c)\ge{}l(\restr{c}_{[0,t_{1}]})\ge\varepsilon/2>0$.\proven
}

\thm{
	$\forall{}p\in{}M$, $\exists{}W$ a neighborhood of $p$, and $\exists\delta>0$ \st{} $\forall{}q\in{}W$,
	\[
		\restr{\exp_{q}}_{B_{\delta}(0)}:B_{\delta}(0)\overset{\sim}{\to}\exp_{q}(B_{\delta}(0))
	\]
	is a diffeomorphism onto its image, and $W\subseteq\exp_{q}(B_{\delta}(0))$.\nn
	Proof: Recall that given $p\in{}M$, there's a neighborhood $V$ of $p$, and $\varepsilon>0$, such that $\exp:B_{\varepsilon}(TV)\to{}M$ is defined, where $B_{\varepsilon}(TV)=\set{(q,v)\in{}TV\mid{}q\in{}V,\norm{v}<\varepsilon}$. In other words, $G(t,q,v)$ is defined past $t=1$. Now, define
	\[
		\map{F:B_{\varepsilon}(TV)}{M\times{}M}{(q,v)}{(q,\exp_{q}(v))}
	\]
	In particular, $F(p,0)=(p,p)$. We claim that $F$ is a local diffeomorphism near $0$. To check this fact, introduce coordinates near $p$, and then ``double'' them to get coordinates on $M\times{}M$ near $(p,p)$. Then the Jacobian is the block matrix
	\[
		dF_{(p,0)}=\begin{pmatrix}
			I & 0\\
			I & I
		\end{pmatrix}
	\]
	with the bottom-left entry being $I$ because $\exp_{q}0=q$, and the bottom right entry being $I$ because $d(\exp_{p})_{0}=\Id$. This matrix is invertible, so $F$ is a local diffeomorphism. Thus, there's a neighborhood $V'$ of $p$, with $V'\subset{}V$, and a $\delta>0$, such that
	\[
		\restr{F}_{B_{\delta}(TV')}\overset{\sim}{\to}F(B_{\delta}(TV'))
	\]
	is a diffeomorphism onto its image, which is a neighborhood of $(p,p)\in{}M\times{}M$. Thus, $\exists{}W$, a neighborhood of $p$, such that $W\times{}W\subset{}F(B_{\delta}(TV'))$. In other words,
	\[
		W\times{}W\subset\set{(q,\exp_{q}(v))\mid{}q\in{}V',\norm{v}<\delta}\overset{F}{\cong}\set{(q,v)\mid{}q\in{}V',\norm{v}<\delta}
	\]
	So $\forall{}q\in{}W$, $\set{q}\times{}B_{\delta}(0)\overset{\sim}{\to}\set{q}\times\exp_{q}(B_{\delta}(0))$ via $F$ under an appropriate restriction.\proven
}

\defn{
	Such a neighborhood $W$ is called a \u{totally normal neighborhood}.\n
}

\newpage
\lemma{
	Let $W$ be a totally normal neighborhood, and $p,q\in{}W$. Then there is a unique geodesic (up to reparameterization) joining $p$ and $q$, and entirely contained in $W$. Moreover, this geodesic is the shortest path (i.e. continuous and piecewise $C^{1}$) joining $p$ to $q$.\nn
	Proof: Let $\gamma:[0,1]\to{}W$ be a geodesic, with $\gamma(0)=p$ and $\gamma(1)=q$. Lift $\gamma$ to $T_{p}M$ by $\restr{\exp_{p}\inv}_{W}$. Then $\gamma(t)=\exp_{p}(c(t))$, where $t\mapsto{}c(t)\in{}T_{p}M$ and $c(0)=0$. Note that $\dot\gamma(0)=\dot{c}(0)$, and $t\mapsto\exp_{p}(t\dot{c}(0))$ is a geodesic with the same initial conditions as $\gamma$. By the uniqueness (up to reparameterization) of geodesics with initial conditions, we must have $\gamma(t)=\exp_{p}(t\dot{c}(0))$, $\forall{}t$. This also implies that $\gamma$ is the shortest path from $p$ to $q$, since it's a radial geodesic.\proven
}

\cor{
	All geodesics are locally length-minimizing.\nn
	Proof: Let $\gamma:I\to{}M$ be a geodesic. Take $t_{0}\in\Int{}I$, i.e., $t_{0}\in(a,b)\subseteq[a,b]\subseteq{}I$. By the existence of totally normal neighborhoods of $\gamma(t_{0})$, if $b-a$ is small enough, then $\gamma([a,b])$ is contained in a totally normal neighborhood of $\gamma(t_{0})$.\proven
}

\cor{
	Suppose $\gamma:[0,1]\to{}M$ is a path (continuous, and piecewise $C^{1}$), and $d(\gamma(0),\gamma(1))=l(\gamma)$. Then $\gamma$ is a geodesic, and in paricular, it's $C^{\infty}$.\nn
	Proof: The idea is that global length minimization leads to local. Let $t_{0}\in[0,1]$. Again find a neighborhood $[a,b]$ of $t_{0}$ such that $\gamma([a,b])$ is contained in a totally normal neighborhood of $\gamma(t_{0})$. Then $\gamma$ must be the shortest path from $\gamma(a)$ to $\gamma(b)$. This means $\restr{\gamma}_{[a,b]}$ must be a geodesic, and $\gamma$ is smooth in a neighborhood of $t_{0}$.\proven
}
