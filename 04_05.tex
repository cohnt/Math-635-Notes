\par\noindent
Let $M\subset\bar{M}$ be a Riemannian submanifold. Continuing from last time:\n

\thm{
	(Gauss' Formula) Given $W,X,Y,Z\in\mf{X}(\bar{M})$ tangent to $M$, $\forall{}p\in{}M$,
	\[
		\bar{R}(W,X,Y,Z)=R(W,X,Y,Z)+\iprod{B(W,Y),B(X,Z)}-\iprod{B(W,Z),B(X,Y)}
	\]
	\up\nn
	Proof: Well, the left hand side is
	\[
		\bar{R}(W,X,Y,Z)=\iprod{(\bar\nabla_{W}\bar\nabla_{X}-\bar\nabla_{X}\bar\nabla_{W}-\bar\nabla_{\comm{W,X}})Y,Z}=\iprod{\bar\nabla_{W}\bar\nabla_{X}Y,Z}+\iprod{-\bar\nabla_{X}\bar\nabla_{W}Y,Z}+\iprod{-\bar\nabla_{\comm{W,X}}Y,Z}
	\]
	We first work with the first term:
	\[
		\iprod{\bar\nabla_{W}\bar\nabla_{X}Y,Z}=\iprod{\bar\nabla_{W}\paren{\nabla_{X}Y+B(X,Y)},Z}=\iprod{\nabla_{W}\nabla_{X}Y,Z}+\iprod{\bar\nabla_{W}B(X,Y),Z}
	\]
	because $\nabla=\bar\nabla^{T}$, and $Z$ is tangent to $M$. The first term will eventually contribute to $R(W,X,Y,Z)$. As for the second term, note that $B(X,Y)$ is normal to $M$ and $Z$ is tangent to $M$, so $\iprod{B(X,Y),Z}=0$. So we can use the compatibility of $\bar\nabla$ with the metric and differentiation (specifically, the product rule) to get
	\[
		\iprod{\bar\nabla_{W}B(X,Y),Z}=-\iprod{B(X,Y),\bar\nabla_{W}Z}=-\iprod{B(X,Y),B(W,Z)}
	\]
	By performing the analogous computations on the second term, we get
	\[
		\iprod{-\bar\nabla_{X}\bar\nabla_{W}Y,Z}=\iprod{-\nabla_{X}\nabla_{W}Y,Z}+\iprod{B(W,Y),B(X,Z)}
	\]
	As for the third term,
	\[
		\iprod{-\bar\nabla_{\comm{W,X}}Y,Z}=\iprod{-\nabla_{\comm{W,X}}Y,Z}
	\]
	\proven
}

\par\noindent
Recall: This implies that if $\bar{M}$ is flat, we have an orthonormal eigenbasis of the shape operator $e_{i}$, with corresponding eigenvalues $\kappa_{i}$. We get $\forall{}i\ne{}j$, $K(e_{i},e_{j})=\kappa_{i}\kappa_{j}$.\n

\par\noindent
Recall: For $M\subseteq\R^{n+1}$ a hypersurface ($\dim{}M=n$), if $M$ is oriented by a unit normal field $m\ni{}p\mapsto{}N_{p}\in(T_{p}M)^{\perp}$, we can interpret the unit normal field as a map $N:M\to{}S^{n}$. $\forall{}p\in{}M$, the shape operator at $p$ with respect to $N_{p}$, denoted $S_{p}$, satisfies $S_{p}=-(dN)_{p}$ (with $N$ as a map). We call $N$ the Gauss spherical map. In this case, we're implicitly identifying $T_{p}M\cong{}T_{p}S^{n}$.\n

\cor{
	For $M\subset\R^{3}$ a surface, $K(p)=\det(dN)_{p}$, where $K(p)$ is the sectional curvature of $M$ at $p$. (Because $\dim{}M=2$, there's only one tangent plane at a given $p\in{}M$.)\n
}

\par\noindent
We want to explore the global implications of this, when $\dim{}M=2$. Let $N=M\to{}S^{2}$.\n

\par\noindent
Observe: If $K(p)\ne{}0$, then $N$ is a local diffeomorphism at $p$. So if $K$ is non-vanishing, and if $M$ is compact and connected, then $N$ is a covering map! Why is this true? Well, fix $q\in{}S^{2}$. Then $N\inv(q)=\bigsqcup_{i=1}^{I}\set{p_{i}}$. (This is finite because $M$ is compact.) $\forall{}i$, $p_{i}$ has a neighborhood $U_{i}$ such that $\restr{N}_{U_{i}}:U_{i}\to{}N(U_{i})=V_{i}$ is a diffeomorphism. Now, let $V=\bigcap_{i=1}^{I}V_{i}$. This is a neighborhood of $q$, and we claim $V$ is evenly covered; to see this, just let $\tilde{U}_{i}=N\inv(V_{i})\cap{}U_{i}$.\n

\cor{
	If $K$ is non-vanishing, and $M$ is compact and oriented, then $M$ is diffeomorphic to $S^{2}$.\n
}

\ex{
	Consider $M$ an ellipsoid, $K>0$.\n
}

\par\noindent
Question: Is there a metric on $S^{2}$ where $K<0$? Answer: No, purely by topology.\n

\thm{
	(Gauss-Bonnet) Let $n=2m$ be even. Let $M\subset\R^{n+1}$ be a compact, oriented, connected manifold. Let $N:M\to{}S^{n}$ be the Gauss map. Define $\mathscr{K}:M\to\R$, the Gaussian curvature, by $N^{*}(dV_{S^{n}})=\mathscr{K}dV_{M}$, where $dV_{S^{n}}$ is the volume form on $S^{n}$ and $dV_{M}$ is the volume form on $M$. Then
	\[
		\int_{M}\mathscr{K}\,dV_{M}=\frac{1}{2}\Vol(S^{n})\chi(M)
	\]
	where $\chi(M)$ is the Euler characteristic of $M$. Also, $\mathscr{K}$ is intrinsic -- it depends only on the induced metric on $M$.\n
}

\defn{
	The \u{Euler characteristic} of $M$ is $\chi(M)=\sum_{i=0}^{n}(-1)^{i}\dim{}H^{i}(M)$.\n
}

\par\noindent
Note: $\ms{K}(p)=\det(dN)_{p}=\prod_{i=1}^{n}\kappa_{i}$. Why is this product intrinsic? Well, since $n=2m$ is even, we can write
\[
	\ms{K}(p)=\prod_{i=1}^{m}\kappa_{2m-1}\kappa_{2m}
\]
$\kappa_{2m-1}\kappa_{2m}$ is a specific sectional curvature, which we know to be intrinsic.\n

\par\noindent
Plan for how we'll prove Gauss-Bonnet:
\begin{enumerate}
	\item Show $\ms{K}$ is intrinsic (for $n\ge{}4$).
	\item Show $\ms{K}dV_{M}=N^{*}(dV_{S^{n}})$. This implies
	\[
		\int_{M}\ms{K}\,dV_{M}=\int_{M}N^{*}(dV_{S^{n}})=\deg(N)\int_{S^{n}}dV_{S^{n}}=\deg(N)\Vol(S^{n})
	\]
	\item $\deg(N)=\frac{1}{2}\chi(M)$
\end{enumerate}
Steps 2 and 3 are just pure differenial topology.\n

\par\noindent
``Degree theory'': $N:M\to{}S^{n}$ induces
\[
	\begin{tikzcd}
		N^{*}\!:{} &[-32.5pt] H^{n}(S^{n}) \arrow[r] \arrow[d,symbol=\cong] &[-10pt] H^{n}(M) \arrow[d, symbol=\cong]\\[-10pt]
		& \R & \R
	\end{tikzcd}
\]
So $N^{*}$ is just multiplication by a number. In fact, that number is an integer, and is called the degree of $N$.\n
