\section*{Space of Constant Curvature}

\par\noindent
Such spaces are locally isometric to a sphere (if the curvature is positive), a Euclidean space (if the curvature is zero), or a hyperbolic space (if the curvature is negative).\n

\lemma{
	$M$ has constant sectional curvature $K_{0}\in\R$ iff $\forall{}X,Y,W,Z\in\mf{X}(M)$,
	\[
		(X,Y,W,Z)=K_{0}(\iprod{Y,W}\iprod{X,Z}-\iprod{X,W}\iprod{Y,Z})
	\]
	Proof: Note that by definition of $K$,
	\[
		(X,Y,Y,X)=K(X,Y)\underbrace{(\iprod{X,X}\iprod{Y,Y}-\iprod{X,Y}^{2})}_{=\abs{X\wedge{}Y}}
	\]
	This is a special case of the identity we want! Proving $\Leftarrow$ is trivial. As for proving $\Rightarrow$, we show that the right hand side has the same symmetry properties as the left hand side. (This is just a messy computation.) Then we use the fact that we know $K$ can determine $\mc{R}$ if it's applied everywhere.\proven
}

\cor{
	Let $M$ have constant curvature, $\gamma$ a geodesic of speed $1$, and $J$ a normal Jacobi field. Then the Jacobi equation becomes $J''+KJ=0$. (Recall the notation: $J'=\frac{DJ}{dt}$, $J''=\frac{D^{2}J}{dt^{2}}$.)\nn
	Proof: By the lemma, we have, $\forall{}X\in\mf{X}(M)$,
	\[
		\iprod{\mc{R}(J,\dot\gamma)\dot\gamma,X}=(J,\dot\gamma,\dot\gamma,X)=K(\norm{\dot\gamma}^{2}\iprod{J,X}-\iprod{J,\dot\gamma}\iprod{\dot\gamma,X})
	\]
	This is equal to $K\iprod{J,X}$ for all $X$, $|mc{R}(J,\dot\gamma)\do\gamma=KJ$.\proven
}

\par\noindent
Now, we consider a generalization of the surface case. Let $\gamma$ be a geodesic on $M$, with constant curvature.\n
Let $N(t)\in\Gamma_{\gamma}(TM)$ be a unit normal and parallel field, determined by $N(0)$. In the surface case, we call this $\dot\gamma^{\perp}$. Define $J(t)=\varphi(t)N(t)$. Then $J$ is Jacobi iff $\ddot\varphi+K\varphi=0$.\n

\cor{
	If $J(0)=0$, then $J(t)=\left\{\begin{array}{ll}
		A\sin(\sqrt{K}t)N & K>0\\
		tN & K=0\\
		A\sinh(\sqrt{-K}t)N & K<0
	\end{array}\right.$\n\n
}

\par\noindent
Recall the differential of the exponenial map:\n

\thm{
	$d(\exp_{p})_{v}(w)=J(1)$, where $J$ is the Jacobi field along $\gamma:t\mapsto\exp_{p}(tv)$ \st{} $J(0)=0$, $J'(0)=w$.\nn
	Proof: Define $f(s,t)=\exp_{p}(t(v+sw))$. Observe: $d(\exp_{p})_{v}(w)=\restr{\d_{s}f}_{s=0,t=1}$. Let $J(t)=\d_{s}f(s=0,t)$. This is a Jacobi field, since the $t$ curves are geodesics. Now, we just need to check that $J$ satisfies the initial conditions.
	\[
		J(0)=0,\qquad{}J'=\frac{D}{dt}\d_{s}f=\frac{D}{ds}\d_{t}f,\qquad{}\d_{t}f(s,t=0)=d(\exp)_{0}(v+sw)=v+sw\in{}T_{p}M
	\]
	$s\mapsto\d_{t}f(s,0)$ is entirely contained in $T_{p}M$. Still, it's a field along $s\mapsto{}f(s,0)=p$, so it's just a constant ``curve''. Finally, $\frac{D}{ds}\d_{t}f=\frac{d}{ds}(v+sw)=w$.\proven
}

\par\noindent
Next, we examine a result of the ``rate of spreading of geodesics''. As before, define $f(s,t)=\exp_{p}(t(v+sw))$.\n

\prop{
	Take $v,w\in{}T_{p}M$ orthonormal, and let $\pi=\vspan(v,w)$, $\gamma(t)=\exp_{p}(tv)$. Let $J(t)$ be the Jacobi field \st{} $J(0)=0$, $J'(0)=w$. (Again, exactly as before.) Then
	\[
		\norm{J(t)}=t-\frac{t^{3}}{6}K_{p}(\pi)+O(t^{3})\qquad\ptxt{as}\qquad{}t\to{}0
	\]
	as $t\to\infty$.\n
}

\par\noindent
Interpretation: Rate of spreading of the ray $t(v+sw)$ with respect to $t\mapsto{}tv$. $\pd{}{s}\abs{t(v+sw)}=tw$. We want to measure the same object on the manifold after exponentiation. We do so with respect to infinitesimal change in $s$.
\[
	\norm{J(t)}\sim\left\{\begin{array}{ll}
		<t & K_{p}(\pi)>0\\
		>t & K_{p}(\pi)<0
	\end{array}\right.
\]

\par\noindent
Proof of the proposition: It's true iff
\[
	\norm{J(t)}^{2}=t^{2}-\frac{t^{4}}{3}K_{p}(\pi)+O(t^{4})
\]
So we need to compute $4$ derivatives of $\iprod{J,J}$ at zero. Define $a_{k}=\iprod{J,J}^{(k)}(t=0)$. Then clearly $a_{0}=0$. $\iprod{J,J}'=2\iprod{J',J}$, so $a_{1}=0$.
\[
	\frac{1}{2}\iprod{J,J}''=\iprod{J'',J}+\iprod{J',J'}\quad\Rightarrow\quad{}a_{2}=\norm{w}^{2}=1
\]
\[
	\frac{1}{2}\iprod{J,J}^{(3)}=\iprod{J^{(3)},J}+\iprod{J'',J'}+2\iprod{J'',J'}=\underbrace{\iprod{J^{(3)},J}}_{=0}+3\underbrace{\iprod{J'',J'}}_{\hspace{-35pt}\mathrlap{=-\iprod{\mc{R}(J,\dot\gamma)\dot\gamma,J'}}}=0
\]
because $J(0)=0$ by the Jacobi equations. Finally, we compute the fourth derivative:
\[
	\frac{1}{2}\iprod{J,J}^{(4)}=\cancelto{0}{\iprod{J^{(4)},J}}+3\left(\vphantom{\iprod{J^{(3)},J'}}\right.\underbrace{\iprod{J^{(3)},J'}}_{?}+\underbrace{\iprod{J'',J''\vphantom{J^{(3)}}}}_{=0}\left.\vphantom{\iprod{J^{(3)},J'}}\right)
\]
We will figure out $\iprod{J^{(3)},J'}$ in the next lecture.
