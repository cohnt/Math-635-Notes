\lemma{
	Let $M$ be a $C^{\infty}$ manifold. Assume $\forall{}p\in{}M$, we have a vector space $\mathcal{E}_{p}$ (of dimension $r$). Let $\mathcal{E}=\bigsqcup_{p\in{}M}\mathcal{E}_{p}$, and $\pi:\mathcal{E}\to{}M$ the natural projection, where $\mathcal{E}_{p}\mapsto{}p\in{}M$. Assume we're given $\set{U_{\alpha}}$, a cover of $M$, plus bijections $\chi_{\alpha}$ such that $\forall\alpha$, the following diagram commutes:
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			\pi\inv(U_{\alpha}) \arrow[rr,"\chi_{\alpha}"] \arrow[dr, "\pi", swap] \&[-10pt] \&[-10pt] U_{\alpha}\times\R^{r} \arrow[dl]\\
			\& U_{\alpha}
		\end{tikzcd}
	\]
	i.e. $\chi_{\alpha}(\mathcal{E}_{p})=\set{p}\times\R^{r}$.\nn
	Observe that this gives us $\forall\alpha,\beta$ \st{} $U_{\alpha}\cap{}U_{\beta}\ne\emptyset$, a map $\tau_{\alpha,\beta}:U_{\alpha}\cap{}U_{\beta}\to\GL(r,\R)$, by:
	\[
		\begin{tikzcd}[ampersand replacement=\&]
			\&[-20pt] \pi\inv(U_{\alpha}\cap{}U_{\beta}) \arrow[dl, "\chi_{\alpha}", swap] \arrow[dr, "\chi_{\beta}"] \&[-20pt]\\
			(U_{\alpha}\cap{}U_{\beta})\times\R^{r} \arrow[rr,"\chi_{\beta}\of\chi_{\alpha}\inv"] \& \& (U_{\alpha}\cap{}U_{\beta})\times\R^{r}\\[-20pt]
			(p,v) \arrow[rr,mapsto] \& \& (p,\tau_{\alpha,\beta}(p)v)\\[-20pt]
			p\in{}U_{\alpha}\cap{}U_{\beta}\\[-20pt]
			v\in\R^{r}
		\end{tikzcd}
	\]
	This mapping is a ``change of trivialization'', like a transition map for vector bundles. The matrix of $\tau_{\alpha,\beta}$ depends on $p$, but it's always a linear map.\nn
	If, $\forall\alpha,\beta$, $\tau_{\alpha,\beta}$ is $C^{\infty}$, then there is a unique topology on $\mathcal{E}$, and a unique differentiable structure, that makes $\pi:\mathcal{E}\to{}M$ a smooth vector bundle, and each $\chi_{\alpha}$ is a diffeomorphism, i.e., a smooth local trivialization.\n
}

\par\noindent
Observe: On triple intersections, $U_{\alpha}\cap{}U_{\beta}\cap{}U_{\gamma}$, $\forall{}p$, one has $\tau_{\alpha,\beta}(p)\tau_{\beta,\gamma}(p)=\tau_{\alpha,\gamma}(p)$. This is called a ``cocycle condition''.\n

\par\noindent
Now, imageine starting with a cover $\set{U_{\alpha}}$ and $C^{\infty}$ maps $\tau_{\alpha,\beta}:U_{\alpha}\cap{}U_{\beta}\to\GL(r,\R)$ satisfying the cocycle condition. Then, if we choose, $\forall{}p$, $\mathcal{E}_{p}=\R^{r}$, and $\chi_{\alpha}=I_{n}$, then we get a vector bundle $\mathcal{E}\to{}M$. So $\set{U_{\alpha};\tau_{\alpha,\beta}}$, called a \v{C}ech cocycle, is all we need to put together a vector bundle.\n

\par\noindent
Now, we apply the lemma to construct new bundles from old ones.\n

\ex{
	Given $\mathcal{E}',\mathcal{E}''\to{}M$ vector bundles, with ranks $r'$ and $r''$, respectively, define, $\forall{}p\in{}M$, $(\mathcal{E}'\oplus\mathcal{E}'')_{p}=\mathcal{E}'_{p}\oplus\mathcal{E}''_{p}$. Consider a cover $\set{U_{\alpha}}$ of $M$, and local trivializations for both $\mathcal{E}'$ and $\mathcal{E}''$. We get, $\forall\alpha$,
	\begin{align*}
		& \chi_{\alpha}':(\pi')\inv(U_{\alpha})\to{}U_{\alpha}\times\R^{r'}\\
		& \chi_{\alpha}'':(\pi'')\inv(U_{\alpha})\to{}U_{\alpha}\times\R^{r''}
	\end{align*}
	Now define $\pi:\mathcal{E}'\oplus\mathcal{E}''\to{}M$, and
	\[
		\map{\chi_{\alpha}:\pi\inv(U_{\alpha})}{U_{\alpha}\times(\R^{r'}\oplus\R^{r''})}{(p,(v',v''))}{(p,(\chi_{\alpha}'(v'),\chi_{\alpha}''(v''))\\ v'\in\mathcal{E}_{p}'\\ v''\in\mathcal{E}_{p}''}
	\]
	We have to check that the $\tau_{\alpha,\beta}$ are smooth. Note that $\tau_{\alpha,\beta}(p)$ will be a block diagonal matrix, and the two diagonal blocks vary smoothly, since $\chi_{\alpha}$ and $\chi_{\beta}$ are diffeomorphisms. Thus, $\mathcal{E}'\oplus\mathcal{E}''\to{}M$ is a natural bundle. This is called the \u{Whitney direct sum} or \u{Whitney sum}.\n
}

\newpage
\par\noindent
Observe:
\begin{itemize}[topsep=0pt, itemsep=0pt]
	\item $\rank(\mathcal{E}'\oplus\mathcal{E}'')=\rank(\mathcal{E}')+\rank(\mathcal{E}'')$
	\item $\dim(\mathcal{E}'\oplus\mathcal{E}'')=r'+r''+n$, where $n$ is the dimension of the total space.
\end{itemize}
\up\n

\par\noindent
Similarly, we can define $\mathcal{E}'\otimes\mathcal{E}''\to{}M$ with fibers $(\mathcal{E}'\otimes\mathcal{E}'')_{p}=\mathcal{E}'_{p}\otimes\mathcal{E}''_{p}$. $\rank(\mathcal{E}'\otimes\mathcal{E}'')=r'\cdot{}r''$, and $\dim(\mathcal{E}'\otimes\mathcal{E}'')=r'\cdot{}r''+n$.\n

\subsection*{Review of Tensor Products}

\defn{
	Let $V,W$ be finite-dimensional vector spaces. Their \u{tensor product} $V\otimes{}W$ is the \u{free vector space} over $V\times{}W$, modulo an equivalence relation, $\sim$. The free vector space over $V\times{}W$ is the set of all formal finite linear combinations of pairs $(v,w)\in{}V\times{}W$, and $\sim$ is defined such that
	\[
		\left\{\begin{array}{l}
			(v_{1}+v_{2},w)\sim(v_{1},w)+(v_{2},w)\\
			(\lambda{}v,w)\sim\lambda(v,w)
		\end{array}\right.
	\]
	Notation: $\forall(v,w)\in{}V\times{}W$, $v\otimes{}w=[(v,w)]$.\n
}

\par\noindent
Claim: If $(e_{1},\ldots,e_{k})$, $(f_{1},\ldots,f_{l})$ are bases of $V$, $W$, then $\set{e_{i}\otimes{}f_{j}\mid{}i\in\set{1,\ldots,k},j\in\set{1,\ldots,l}}$ is a basis of $V\otimes{}W$.\n

\cor{
	$\dim(V\otimes{}W)=\dim(V)\cdot\dim(W)$.\n
}

\par\noindent
The universal property of $V\otimes{}W$: We have a bilinear map
\[
	\map{V\times{}W}{V\otimes{}W}{(v,w)}{v\otimes{}w}
\]
It's ``universal'' in that for any bilinear map, there's a unique linear map such that the following diagram commutes:
\[
	\begin{tikzcd}
		V\times{}W \arrow[dr,"\ptxt{bilinear map}", swap] \arrow[rr, "(v{,}w)\mapsto{}v\otimes{}w"] &[-10pt] &[-10pt] V\otimes{}W \arrow[dl, "\ptxt{linear map}"]\\
		& Z
	\end{tikzcd}
\]

\par\noindent
There are other realizations of $V\otimes{}W$.
\[
	V\otimes{}W\cong\Hom(V^{*},W)\qquad\ptxt{where}\qquad{}v\otimes{}w\mapsto\paren{\begin{array}{rcl}
		V^{*}\!\!\!\! & \to\!\!\!\! & W\\
		\alpha\!\!\!\! & \mapsto\!\!\!\! & \alpha(v)w
	\end{array}}
\]
This is completely natural -- we don't need to choose a basis.\n

\cor{
	$V^{*}\otimes{}V^{*}\cong\Hom((V^{*})^{*},V^{*})=\Hom(V,V^{*})\overset{\mathclap{\ptxt{claim}}}{\cong}\Bil(V\times{}V,\R)$, by
	\[
		\begin{array}{r>{\!\!\!\!}c>{\!\!\!\!}l}
			\Hom(V,V^{*}) & \cong & \Bil(V\times{}V,\R)\\
			(f:V\to{}V^{*}) & \mapsto & \paren{\begin{array}{r>{\!\!\!\!}c>{\!\!\!\!}l}
				V\times{}V & \to & \R\\
				(v_{1},v_{2}) & \mapsto & (f(v_{1}))(v_{2})
			\end{array}}
		\end{array}
	\]
	(Note that $f(v_{1})\in{}V^{*}$.)\nn
	Thus, $V\otimes{}V\cong\Bil(V^{*}\times{}V^{*},\R)$. This is the realization we will use!
}

\par\noindent
Note: When taking multiple tensor products:
\[
	\underbrace{V\otimes\cdots\otimes{}V}_{k}\otimes\underbrace{V^{*}\otimes\cdots\otimes{}V^{*}}_{l}\cong\{\ptxt{multilinear maps }\underbrace{V^{*}\times\cdots\times{}V^{*}}_{k}\times\underbrace{V\times\cdots\times{}V}_{l}\to\R\}
\]
Now, we put everything together, and define tensor bundles.\n

\defn{
	Given a smooth manifold $M$, $\forall{}p\in{}M$, we define
	\[
		T^{(k,l)}(T_{p}M)=\underbrace{T_{p}M\otimes\cdots\otimes{}T_{p}M}_{k}\otimes\underbrace{T_{p}^{*}M\otimes\cdots\otimes{}T_{p}^{*}M}_{l}
	\]
	$k$ is called the \u{contravariant degree}, and $l$ is called the \u{covariant degree}. Using the lemma from before, we get \u{tensor bundles}
	\[
		\bigsqcup_{p\in{}M}T^{(k,l)}(T_{p}M)=T^{(k,l)}(TM)\to{}M
	\]
}

\exer{
	Compute the rank of this bundle.\n
}

\par\noindent
In coordinate $(x^{1},\ldots,x^{n})$ on $U\osubseteq{}M$, we have a moving frame of $T^{(k,l)}(TM)$:
\[
	\set{\pd{}{x^{i_{1}}}\otimes\cdots\otimes\pd{}{x^{i_{k}}}\otimes{}dx^{j_{1}}\otimes\cdots\otimes{}dx^{j_{l}}\setmid{}i_{a},j_{b}\in\set{1,\ldots,n}}
\]

\defn{
	Any smooth section of a tensor bundle is called a \u{tensor}.\n
}

\par\noindent
Note that any tensor is a combination of these basis elements with $C^{\infty}$ functions as coefficients:
\[
	\sum{}A^{i_{1}\cdots{}i_{k}}_{j_{1}\cdots{}j_{k}}\pd{}{x^{i_{1}}}\otimes\cdots\otimes\pd{}{x^{i_{k}}}\otimes{}dx^{j_{1}}\otimes\cdots\otimes{}dx^{j_{l}}
\]

\defn{
	A \u{Riemannian metric} is a symmetric, positive-definite $(0,2)$ tensor $g_{p}:T_{p}M\times{}T_{p}M\to\R$.\n
}
