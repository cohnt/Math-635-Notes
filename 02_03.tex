\par\noindent
Review: Connections on Vector Bundles\n

\par\noindent
Given $\mathcal{E}\to{}M$ a vector bundle, a connection is an operator
\[
	\map{\nabla:\mathfrak{X}(M)\times\Gamma(\mathcal{E})}{\Gamma(\mathcal{E})}{(X,s)}{\nabla_{X}s}
\]
with universal quantifiers $\forall{}X,Y\in\mathfrak{X}(M)$, $s,t\in\Gamma(\mathcal{E})$, and $f\in{}C^{\infty}(M)$:
\begin{enumerate}[itemsep=0pt, label=(\arabic*)]
	\item $\nabla_{X+Y}s=\nabla_{X}s+\nabla_{Y}s$
	\item $\nabla_{fX}s=f\nabla_{X}s$
	\item $\nabla_{X}(s+t)=\nabla_{X}s+\nabla_{X}t$
	\item $\nabla_{X}(fs)=f\nabla_{X}s+X(f)s$
\end{enumerate}
Properties (1) and (2) together are written as $\nabla$ is linear in $X$ over $C^{\infty}(M)$. Note that property (4) implies that $\forall{}c\in\R$, $\nabla_{X}(cs)=c\nabla_{X}s$.\n

\par\noindent
Last time, we saw that $(\nabla_{X}s)(p)$ only depends on $X_{p}\in{}T_{p}M$ and $\restr{s}{U}$ for any (arbitrarily small) neighborhood $U$ of $p$. In other words, if we fix a section $s\in\Gamma(\mathcal{E})$, $\nabla_{\cdot}s$ is a $1$-form that takes values in sections. And if we fix $X\in\mathfrak{X}(M)$, then $\nabla_{X}\cdot:\Gamma(\mathcal{E})\to\Gamma(\mathcal{E})$ is a derivation.\n

\ex{
	Say $\mathcal{E}=M\times\R^{r}\to{}M$. (This is locally the general case.) Then $\Gamma(M\times\R^{r})\ni{}s\leftrightarrow{}\vec{f}:M\to\R^{r}$ by $\forall{}p\in{}M$, $s(p)=(p,\vec{f}(p))$. This is the same as having a global moving frame $\Gamma(M\times\R^{r})\ni{}E_{i}\leftrightarrow{}\vec{f}_{i}(p)=(0,\ldots,1,\ldots,0)$ (with a $1$ in the $i$th entry), for $i=1,\ldots,r$, because $s=\sum_{i=1}^{r}f^{i}E_{i}$ for $f^{i}\in{}C^{\infty}(M,\R)$, where $\vec{f}=(f^{1},\ldots,f^{r})$.\nn
	Suppose we have a connection $\nabla$. Last time, we defined $\vartheta=(\theta_{i}^{j})$, the connection matrix associated with $(E_{1},\ldots,E_{r})$, by $\forall{}X$, $\forall{}j$,
	\[
		\nabla_{X}E_{j}=\sum_{i=1}^{r}\theta_{i}^{j}(X)E_{j}
	\]
	$\forall{}i,j$, $\theta_{i}^{j}\in\Omega^{1}(M)$ (a $C^{\infty}$ one-form). Then:
	\[
		\nabla_{X}s=\sum_{i=1}^{r}\nabla_{X}(f^{i}E_{i})=\sum_{i=1}^{r}X(f^{i})E_{i}+\sum_{i=1}^{r}f^{i}\nabla_{X}E_{i}=\sum_{i=1}^{r}X(f^{i})E_{i}+\sum_{i,j=1}^{r}f^{i}\theta_{i}^{j}(X)E_{j}
	\]
	The corresponding vector-valued function on $M$ is
	\[
		\nabla_{X}\vec{f}=\paren{X(f^{1})+\sum_{i=1}^{r}\theta_{i}^{1}(X)f^{i},\ldots,X(f^{r})+\sum_{i=1}^{r}\theta_{i}^{r}(X)f^{i}}=\paren{X(f^{1}),\ldots,X(f^{r})}+\paren{\sum_{i=1}^{r}\theta_{i}^{1}(X)f^{i},\ldots,\sum_{i=1}^{r}\theta_{i}^{r}(X)f^{i}}
	\]
	So in vector/matrix notation, with $\vec{f}$ as a column vector, we write
	\[
		\nabla_{X}\vec{f}=d\vec{f}(X)+\vartheta(X)\vec{f}
	\]
	where $\vartheta(X)=(\theta_{i}^{j}(X))$ with lower index $i$ being the columns, and upper index $j$ being the rows.\nn
	Conversely, here, $\vartheta$ can be any $r\times{}r$ matrix of one-forms, and this can be used to define a connection on the trivial bundle!\n
}

\par\noindent
Observe: The previous calculation is valid locally, given some moving frame $(E_{1},\ldots,E_{r})$ of $\mathcal{E}\to{}M$ on $U\subseteq{}M$. Suppose $(F_{1},\ldots,F_{r})$ is another moving frame on $U$. Then $\forall{}i$, $F_{i}=\sum_{j}a_{i}^{j}E_{j}$, where the matrix $A=(a_{i}^{j})$ is invertible at each $p\in{}U$, and $\forall{}i,j$, $a_{i}^{j}\in{}C^{\infty}(U)$. Given $\nabla$, we get $\vartheta$, the connection matrix corresponding to the $E_{j}$'s, and $\tilde\vartheta$, the connection matrix corresponding to the $F_{j}$'s.\n

\exer{
	Check that $\tilde\vartheta=A\inv{}dA+A\inv\vartheta{}A$.\n
}

\par\noindent
A special feature of the case where $\mathcal{E}=TM$ ($r=n=\dim{}M$). Consider again a moving frame $(E_{1},\ldots,E_{n})$. In this case, we can write $X=\sum_{k}a^{k}E_{k}$. In the generic case,
\[
	\nabla_{X}E_{i}=\sum_{j}\theta_{i}^{j}(X)E_{j}=\sum_{j,k}a^{k}\underbrace{\theta_{i}^{j}(E_{k})}_{\mathclap{\substack{\ptxt{Christoffel}\\ \ptxt{Symbols}}}}E_{j}
\]

\defn{
	$\forall{}i,j,k$, $\Gamma_{ki}^{j}\eqdef\theta_{i}^{j}(E_{k})\in{}C^{\infty}(U)$ are the \u{Christoffel symbols}.\n
}

\par\noindent
Note that $\forall{}i,k$, we get $\nabla_{E_{k}}E_{i}=\sum_{j}\Gamma_{ki}^{j}E_{j}$. So $\Gamma_{ki}^{j}$ determines $\vartheta$, and therefore $\nabla$ on $U$.\n

\par\noindent
Now, back to the general case: vector bundle $\mathcal{E}\to{}M$ with connection $\nabla$. We want to look at ``parallelism'' and ``parallel transport''.\n

\defn{
	Let $\gamma:[a,b]\to{}M$, $s\in\Gamma(\mathcal{E})$. Then $s$ is said to be \u{covariant constant}, or \u{parallel}, iff $\forall{}t\in[a,b]$, $(\nabla_{\dot\gamma(t)}s)(\gamma(t))=0$.\n
}

\par\noindent
Let's analyze this equation... It will turn out to be a system of ordinary differential equations!\n

\par\noindent
Let $(E_{1},\ldots,E_{r})$ be a moving frame on $U$, a neighborhood of $\gamma(t)$. For the time being, just assume $\Im(\gamma)\subseteq{}U$. We can write $s=\sum_{i}f^{i}E_{i}$, for some $f^{i}\in{}C^{\infty}(U)$. Let $\vartheta$ be the connection matrix w.r.t. the $E_{i}$'s.\n
Introduce $\vec{f}=(f^{1},\ldots,f^{r}):U\to\R^{r}$. We saw that $\nabla_{\dot\gamma}\vec{f}=d\vec{f}(\dot\gamma)+\vartheta(\dot\gamma)\vec{f}$. Define $f^{i}(t)=f^{i}(\gamma(t))$, so here, $\vec{f}(t)=\vec{f}(\gamma(t))$. Then
\[
	d\vec{f}(\dot\gamma)=\frac{d}{dt}\vec{f}\qquad\ptxt{and}\qquad\frac{d}{dt}\vec{f}+\vartheta(\dot\gamma)\vec{f}=\vec{0}
\]
(taking $\vec{f}$ to be the column vector of the $f^{i}(t)$'s, and $\vartheta(\dot\gamma)$ being a $t$-dependent matrix).
