\subsection*{Hamiltonian Formulation of Geodesic Flow}

\defn{
	A \u{symplectic manifold} is a pair $(X,\omega)$ where $\omega$ is a $2$-form on $X$ \st{} $d\omega=0$ and $\omega$ is pointwise non-denerate: $\forall{}m\in{}X$, the map
	\[
		\map{\omega_{m}^{\sharp}:T_{m}X}{T_{m}^{*}X}{v}{-\omega_{m}(\cdot,v)}
	\]
	is an isomorphism.\n
}

\par\noindent
Note: This implies the dimension of $X$ is even, since skew symmetric forms on odd dimensional spaces are singular.\n

\ex{
	Let $X=\R^{2n}$, with coordinates $(x^{1},\ldots,x^{n},p_{1},\ldots,p_{n})$. Then $\omega=\sum_{i}dp_{i}\wedge{}dx^{i}$, as a matrix, is $\smallPMatrix{0 & I_{n}\\ -I_{n} & 0}$.\n
}

\par\noindent
We will see, $\forall{}M$ smooth manifolds, that $X=T^{*}M$ is naturally a symplectic manifold. And on symplectic manifolds, we can define Hamiltonian dynamics.\n

\defn{
	Given $H\in{}C^{\infty}(X)$, where $(X,\omega)$ is a symplectic manifold, the \u{Hamilton field} of $H$, $\Xi_{H}\in\mf{X}(X)$, is defined by the condition that $-\iota_{\Xi_{H}}=\omega(\cdot,\Xi_{H})=dH$. Existence is guaranteed by the non-degeneracy of $\omega$.\n
}

\par\noindent
We want to compute a local formula: In $\R^{2n}$, $H:\R^{2n}\to\R$. Then $\Xi_{H}=\pd{H}{p_{i}}\pd{}{x^{i}}-\pd{H}{x^{i}}\pd{}{p_{i}}$. The flow/trajectory of $\Xi_{H}$ are
\[
	\left\{\begin{array}{ll}
		\dot{x}^{i}(t)=\pd{H}{p_{i}}(x(t),p(t))\\
		\dot{p}_{i}(t)=-\pd{H}{x^{i}}(x(t),p(t))
	\end{array}\right.
\]
These are Hamilton's equations.\n

\exer{
	$H=\frac{1}{2m}\norm{p}^{2}+V(x)$ is Newton's second law, $\ddot{x}=-\del{}V$.\n
}

\par\noindent
Properties of Hamiltonian flows (i.e. flows of $\Xi_{H}$):
\begin{enumerate}[label=(\alph*)]
	\item $\Xi_{H}(H)=0$, i.e., $H$ is constant along trajectories of $\Xi_{H}$.\n
	Proof: $\Xi_{H}(H)=dH(\Xi_{H})=\omega(\Xi_{H},\Xi_{H})=0$ by antisymmetry.\proven
	\item $\mathcal{L}_{\Xi_{H}}\omega=0$.\n
	Proof: Use Cartan's formula. $\mc{L}_{\Xi_{H}}\omega=\iota_{\Xi_{H}}\underbrace{d\omega}_{=0}+d(\underbrace{\iota_{\Xi_{H}}\omega}_{=-dH})=-d^{2}H=0$.\proven
\end{enumerate}

\subsubsection*{Volume elements (Liouville)}

\par\noindent
On any symplectic manifold $(X,\omega)$, the form $\frac{\omega^{n}}{n!}$ is a volume form.\n

\ex{
	In $\R^{2n}$, $\frac{\omega^{n}}{n!}=dp_{1}\wedge{}dx^{1}\wedge{}dp_{2}\wedge{}dx^{2}\wedge\cdots\wedge{}dp_{n}\wedge{}dx^{n}$.\n
}

\par\noindent
If we're given a Hamiltonian $H\in{}C^{\infty}(X)$, and $c\in\R$ is a regular value of $H$, then let $\Sigma=H\inv(c)\hookrightarrow{}X$, a $\codim$-$1$ submanifold. We claim that $\exists\unique\lambda\in\Omega^{2n-1}(X)$ \st{} in a neighborhood of $\Sigma$, $\frac{\omega^{n}}{n!}=\lambda\wedge{}dH$, and $\iota^{*}(\omega)$ is unique. This is a volume form on $\Sigma$.\n

\cor{
	The Hamilton flow of $H$ preserves $\frac{\omega^{n}}{n!}$, and its restriction to any regular level set $H\inv(c)$ preserves the Liouville measure on that level set.
	\[
		\phi_{t}^{*}\paren{\frac{\omega^{n}}{n!}}=\frac{\omega^{n}}{n!};\qquad\paren{\restr{\phi_{t}}_{H\inv(c)}}^{*}(\iota^{*}\lambda)=(\iota^{*}\lambda)
	\]
	\n
}

\subsection*{Symmetries}

\par\noindent
Question: Given $H,G\in{}C^{\infty}(X)$, the Hamiltonian flows of $H$ and $G$ commute iff $\comm{\Xi_{G},\Xi_{H}}=0$, which is true iff $\Xi_{H}$ is $\phi_{t}^{G}$-related to itself.\n

\lemma{
	For $X$ connected, this is equivalent to $(\phi_{t}^{G})^{*}dH=dH$.\nn
	Proof:
	\begin{align*}
		(\phi_{t}^{G})^{*}dH=dH & \Leftrightarrow \mc{L}_{\Xi_{G}}(dH)=0\\
		& \Leftrightarrow \cancel{\iota_{\Xi_{G}}d^{2}H}+d(\iota_{\Xi_{G}}dH)=dH(\Xi_{G})=0\\
		& \Leftrightarrow d(dH(\Xi_{G}))=0\\
		& \Leftrightarrow dH(\Xi_{G})=\Xi_{G}(H)\;\ptxt{is locally constant}\\
		& \Leftrightarrow \Xi_{G}(H)\;\ptxt{is constant (because $X$ is connected)}
	\end{align*}
	This is a symmetric condition:
	\[
		dH(\Xi_{G})=\omega(\Xi_{G},\Xi_{H})=-\omega(\Xi_{H},\Xi_{G})=-dG(\Xi_{H})
	\]
	\proven
}

\par\noindent
The main example of $(X,\omega)$ is $T^{*}M$, for some arbitrary smooth manifold $M$.\n

\prop{
	For an arbitrary smooth manifold $M$, $T^{*}M$ has a natural symplectic structure.\nn
	Proof: We'll show that $T^{*}M$ has a natural, ``tautological'' $1$-form, $\alpha$, which is sometimes called a Liouville form, defined by:\n
	For $(x,\xi)\in{}T^{*}M$, $x\in{}M$, $\xi\in{}T_{x}^{*}M$, let $v\in{}T_{(x,\xi)}(T^{*}M)$. Then $\alpha_{(x,\xi)}(v)=\xi(\underbrace{\pi_{*}(v)}_{\in{}T_{x}M})$. In coordinates, say we have $(x^{1},\ldots,x^{n})$ on $U\subset{}M$, and $(x^{1},\ldots,x^{n},p_{1},\ldots,p_{n})$ on $T{*}U$. Then $\xi=p_{i}(\xi)dx^{i}$, so $v=a^{i}\pd{}{x^{i}}+b_{i}\pd{}{p_{i}}$. $\pi_{*}(v)=a^{i}\pd{}{x^{i}}$. $\xi(\pi_{*}(v))=p_{i}(\xi)a^{i}$. Altogether, $\alpha=p_{i}dx^{i}$, and $\omega+d\alpha=dp_{i}\wedge{}dx^{i}$, just as in $\R^{2n}$.
}
